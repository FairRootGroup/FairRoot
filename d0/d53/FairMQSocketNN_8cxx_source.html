159 int64_t FairMQSocketNN::Send(const vector<unique_ptr<FairMQMessage>>& msgVec, const int flags)
160 {
161 #ifdef MSGPACK_FOUND
162  // create msgpack simple buffer
163  msgpack::sbuffer sbuf;
164  // create msgpack packer
165  msgpack::packer<msgpack::sbuffer> packer(&sbuf);
166 
167  // pack all parts into a single msgpack simple buffer
168  for (int i = 0; i < msgVec.size(); ++i)
169  {
170  static_cast<FairMQMessageNN*>(msgVec[i].get())->fReceiving = false;
171  packer.pack_bin(msgVec[i]->GetSize());
172  packer.pack_bin_body(static_cast<char*>(msgVec[i]->GetData()), msgVec[i]->GetSize());
173  }
174 
175  int64_t nbytes = nn_send(fSocket, sbuf.data(), sbuf.size(), flags);
176  if (nbytes >= 0)
177  {
178  fBytesTx += nbytes;
179  ++fMessagesTx;
180  return nbytes;
181  }
182  if (nn_errno() == EAGAIN)
183  {
184  return -2;
185  }

250 int64_t FairMQSocketNN::Receive(vector<unique_ptr<FairMQMessage>>& msgVec, const int flags)
251 {
252 #ifdef MSGPACK_FOUND
253  // Warn if the vector is filled before Receive() and empty it.
254  if (msgVec.size() > 0)
255  {
256  LOG(WARN) << "Message vector contains elements before Receive(), they will be deleted!";
257  msgVec.clear();
258  }
259 
260  // pointer to point to received message buffer
261  char* ptr = NULL;
262  // receive the message into a buffer allocated by nanomsg and let ptr point to it
263  int nbytes = nn_recv(fSocket, &ptr, NN_MSG, flags);
264  if (nbytes >= 0) // if no errors or non-blocking timeouts
265  {
266  // store statistics on how many bytes received
267  fBytesRx += nbytes;
268  // store statistics on how many messages received (count messages instead of parts)
269  ++fMessagesRx;
270 
271  // offset to be used by msgpack to handle separate chunks
272  size_t offset = 0;
273  while (offset != nbytes) // continue until all parts have been read
274  {
275  // vector of chars to hold blob (unlike char*/void* this type can be converted to by msgpack)
276  std::vector<char> buf;
277 
278  // unpack and convert chunk
279  msgpack::unpacked result;
280  unpack(result, ptr, nbytes, offset);
281  msgpack::object object(result.get());
282  object.convert(buf);
283  // get the single message size
284  size_t size = buf.size() * sizeof(char);
285  unique_ptr<FairMQMessage> part(new FairMQMessageNN(size));
286  static_cast<FairMQMessageNN*>(part.get())->fReceiving = true;
287  memcpy(part->GetData(), buf.data(), size);
288  msgVec.push_back(move(part));
289  }
290 
291  nn_freemsg(ptr);
292  return nbytes;
293  }
294  if (nn_errno() == EAGAIN)
295  {
296  return -2;
297  }
298  if (nn_errno() == ETERM)
299  {
300  LOG(INFO) << "terminating socket " << fId;
301  return -1;
302  }
303  LOG(ERROR) << "Failed receiving on socket " << fId << ", reason: " << nn_strerror(errno);
304  return nbytes;
305 #else /*MSGPACK_FOUND*/
306  LOG(ERROR) << "Cannot use nanomsg multipart because MessagePack was not found.";
307  exit(EXIT_FAILURE);
308 #endif /*MSGPACK_FOUND*/
309 }
