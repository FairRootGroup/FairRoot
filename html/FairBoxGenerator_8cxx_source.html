137 Bool_t FairBoxGenerator::Init()
138 {
139  // Initialize generator
140 
141  // Check for particle type
142  TDatabasePDG* pdgBase = TDatabasePDG::Instance();
143  TParticlePDG* particle = pdgBase->GetParticle(fPDGType);
144 
145  if (! particle) {
146  Fatal("FairBoxGenerator","PDG code %d not defined.",fPDGType);
147  } else {
148  fPDGMass = particle->Mass();
149  }
150 
151  if (fPhiMax-fPhiMin>360){
152  Fatal("Init()","FairBoxGenerator: phi range is too wide: %f<phi<%f",
153  fPhiMin,fPhiMax);
154  }
155  if (fEkinRangeIsSet){
156  if (fPRangeIsSet){
157  Fatal("Init()","FairBoxGenerator: Cannot set P and Ekin ranges simultaneously");
158  } else {
159  // Transform EkinRange to PRange, calculate momentum in GeV, p = √(K² + 2Kmc²)
160  fPMin = TMath::Sqrt(fEkinMin*fEkinMin + 2*fEkinMin*fPDGMass);
161  fPMax = TMath::Sqrt(fEkinMax*fEkinMax + 2*fEkinMax*fPDGMass);
162  fPRangeIsSet = kTRUE;
163  fEkinRangeIsSet = kFALSE;
164  }
165  }
166  if (fPRangeIsSet && fPtRangeIsSet) {
167  Fatal("Init()","FairBoxGenerator: Cannot set P and Pt ranges simultaneously");
168  }
169  if (fPRangeIsSet && fYRangeIsSet) {
170  Fatal("Init()","FairBoxGenerator: Cannot set P and Y ranges simultaneously");
171  }
172  if ( (fThetaRangeIsSet && fYRangeIsSet) ||
173  (fThetaRangeIsSet && fEtaRangeIsSet) ||
174  (fYRangeIsSet && fEtaRangeIsSet) ) {
175  Fatal("Init()","FairBoxGenerator: Cannot set Y, Theta or Eta ranges simultaneously");
176  }
177  if (fPointVtxIsSet && fBoxVtxIsSet) {
178  Fatal("Init()","FairBoxGenerator: Cannot set point and box vertices simultaneously");
179  }
180 
181 
182  return kTRUE;
183 }

186 Bool_t FairBoxGenerator::ReadEvent(FairPrimaryGenerator* primGen)
187 {
188  // Generate one event: produce primary particles emitted from one vertex.
189  // Primary particles are distributed uniformly along
190  // those kinematics variables which were limitted by setters.
191  // if SetCosTheta() function is used, the distribution will be uniform in
192  // cos(theta)
193 
194  Double32_t pabs=0, phi, pt=0, theta=0, eta, y, mt, px, py, pz=0;
195 
196  // Generate particles
197  for (Int_t k = 0; k < fMult; k++) {
198  phi = gRandom->Uniform(fPhiMin,fPhiMax) * TMath::DegToRad();
199 
200  if (fPRangeIsSet ) { pabs = gRandom->Uniform(fPMin,fPMax); }
201  else if (fPtRangeIsSet) { pt = gRandom->Uniform(fPtMin,fPtMax); }
202 
203  if (fThetaRangeIsSet) {
204  if (fCosThetaIsSet)
205  theta = acos(gRandom->Uniform(cos(fThetaMin* TMath::DegToRad()),
206  cos(fThetaMax* TMath::DegToRad())));
207  else {
208  theta = gRandom->Uniform(fThetaMin,fThetaMax) * TMath::DegToRad();
209  }
210  } else if (fEtaRangeIsSet) {
211  eta = gRandom->Uniform(fEtaMin,fEtaMax);
212  theta = 2*TMath::ATan(TMath::Exp(-eta));
213  } else if (fYRangeIsSet) {
214  y = gRandom->Uniform(fYMin,fYMax);
215  mt = TMath::Sqrt(fPDGMass*fPDGMass + pt*pt);
216  pz = mt * TMath::SinH(y);
217  }
218 
219  if (fThetaRangeIsSet || fEtaRangeIsSet) {
220  if (fPRangeIsSet ) {
221  pz = pabs*TMath::Cos(theta);
222  pt = pabs*TMath::Sin(theta);
223  } else if (fPtRangeIsSet) {
224  pz = pt/TMath::Tan(theta);
225  }
226  }
227 
228  px = pt*TMath::Cos(phi);
229  py = pt*TMath::Sin(phi);
230 
231  if (fBoxVtxIsSet) {
232  fX = gRandom->Uniform(fX1,fX2);
233  fY = gRandom->Uniform(fY1,fY2);
234  }
235 
236  if (fDebug)
237  printf("BoxGen: kf=%d, p=(%.2f, %.2f, %.2f) GeV, x=(%.1f, %.1f, %.1f) cm\n",
238  fPDGType, px, py, pz, fX, fY, fZ);
239 
240  primGen->AddTrack(fPDGType, px, py, pz, fX, fY, fZ);
241  }
242  return kTRUE;
243 
244 }
