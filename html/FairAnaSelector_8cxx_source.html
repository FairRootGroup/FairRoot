36 void FairAnaSelector::Init(TTree* tree)
37 {
38  // The Init() function is called when the selector needs to initialize
39  // a new tree or chain. Typically here the branch addresses and branch
40  // pointers of the tree will be set.
41  // It is normally not necessary to make changes to the generated
42  // code, but the routine can be extended by the user if needed.
43  // Init() will be called many times when running on PROOF
44  // (once per file to be processed).
45  if (!tree) {
46  LOG(warn) << "FairAnaSelector::Init(): There is no tree.";
47  return;
48  } else {
49  LOG(info) << "FairAnaSelector::Init(): Got tree : \"" << tree << "\"";
50  LOG(info) << "FairAnaSelector::Init(): Tree name : \"" << tree->GetName() << "\"";
51  LOG(info) << "FairAnaSelector::Init(): Tree title : \"" << tree->GetTitle() << "\"";
52  LOG(info) << "FairAnaSelector::Init(): Tree filename: \"" << tree->GetCurrentFile()->GetName() << "\"";
53  }
54 
55  if ( fRunAna ) {
56  LOG(info) << "FairAnaSelector::Init(): Already have fRunAna.";
57 
58  LOG(info) << "FairAnaSelector::Init(): SetInTree(" << tree << ")";
59  fProofSource->SetInTree(tree);
60  LOG(info) << "FairAnaSelector::Init(): SetInTree done";
61 
62  LOG(info) << "FairAnaSelector::Init(): Containers static? " << (fRunAna->GetContainerStatic()?"YES":"NO");
63 
64  if ( !fRunAna->GetContainerStatic() ) {
65  fRunAna->InitContainers();
66  }
67  FairRootManager::Instance()->UpdateBranches();
68  } else {
69  LOG(info) << "FairAnaSelector::Init(): Have to create fRunAna.";
70 
71  TString vmcPath = gSystem->Getenv("VMCWORKDIR");
72 
73  TNamed* contStat = static_cast<TNamed*>( fInput->FindObject("FAIRRUNANA_fContainerStatic") );
74  TNamed* outStat = static_cast<TNamed*>( fInput->FindObject("FAIRRUNANA_fProofOutputStatus") );
75  TNamed* outFile = static_cast<TNamed*>( fInput->FindObject("FAIRRUNANA_fOutputFileName") );
76  TNamed* outDir = static_cast<TNamed*>( fInput->FindObject("FAIRRUNANA_fOutputDirectory") );
77  TNamed* par1Name = static_cast<TNamed*>( fInput->FindObject("FAIRRUNANA_fParInput1FName") );
78  TNamed* par2Name = static_cast<TNamed*>( fInput->FindObject("FAIRRUNANA_fParInput2FName") );
79  TString containerS = contStat->GetTitle();
80  TString outputStat = outStat->GetTitle();
81  TString par1Str = par1Name->GetTitle();
82  TString par2Str = par2Name->GetTitle();
83  TString outFileName = outFile->GetTitle();
84  TString outDirName = outDir->GetTitle();

86  LOG(info) << "FairAnaSelector::Init(): out status : \"" << outputStat.Data() << "\"";

88  LOG(info) << "FairAnaSelector::Init(): par2 file : \"" << par2Str.Data() << "\"";
89 
90  LOG(info) << "FairAnaSelector::Init(): OutputFile option \"" << outputStat.Data() << "\" RECOGNIZED";
91 
92  if ( outputStat.Contains("copy") ) {
93  TString outputFileName = outFile->GetTitle();
94  if ( outputFileName[0] != '/' ) {
95  outputFileName = Form("%s/%s",outDir->GetTitle(),outFile->GetTitle());
96  }
97  outputFileName.Remove(outputFileName.Length()-5);
98  outputFileName = Form("%s_worker_%s.root",outputFileName.Data(),gProofServ->GetOrdinal());
99  // outputFileName = outputFileName(outputFileName.Last('/')+1,outputFileName.Length());
100  LOG(info) << "the file name = \"" << outputFileName.Data() << "\"";
101  fFile = TFile::Open(outputFileName.Data(),"RECREATE");
102  }
103  else if ( outputStat.Contains("merge") ) {
104  TString outputFileName = outFile->GetTitle();
105  if ( outputFileName[0] != '/' ) {
106  outputFileName = Form("%s/%s",outDir->GetTitle(),outFile->GetTitle());
107  }
108  outputFileName = outputFileName(outputFileName.Last('/')+1,outputFileName.Length());
109  fProofFile = new TProofOutputFile(outputFileName.Data());
110  if (!(fFile = fProofFile->OpenFile("RECREATE"))) {
111  Warning("SlaveBegin", "problems opening file: %s/%s",
112  fProofFile->GetDir(), fProofFile->GetFileName());
113  }
114  }
115 
116  fRunAna = new FairRunAnaProof("RunOnProofWorker");
117 
118  LOG(info) << "FairAnaSelector::Init(): SetInTree(" << tree << ")";
119  fProofSource = new FairFileSource(tree->GetCurrentFile());
120  fRunAna->SetSource(fProofSource);
121  LOG(info) << "FairAnaSelector::Init(): SetInTree done";
122 
123  fRunAna->SetSink(new FairRootFileSink(fFile));
124  if ( containerS == "kTRUE" ) {
125  fRunAna->SetContainerStatic(kTRUE);
126  } else {
127  fRunAna->SetContainerStatic(kFALSE);
128  }
129 
130  // ----- Parameter database --------------------------------------------
131  FairRuntimeDb* rtdb = fRunAna->GetRuntimeDb();
132 
133  if ( par1Str.Contains(".root") ) {
134  FairParRootFileIo* parInput1 = new FairParRootFileIo();
135  parInput1->open(par1Str.Data());
136  rtdb->setFirstInput (parInput1);
137  }
138  if ( par1Str.Contains(".par") ) {
139  FairParAsciiFileIo* parInput1 = new FairParAsciiFileIo();
140  parInput1->open(par1Str.Data(),"in");
141  rtdb->setFirstInput (parInput1);
142  }
143 
144  if ( par2Str.Contains(".root") ) {
145  FairParRootFileIo* parInput2 = new FairParRootFileIo();
146  parInput2->open(par2Str.Data());
147  rtdb->setSecondInput(parInput2);
148  }
149  if ( par2Str.Contains(".par") ) {
150  FairParAsciiFileIo* parInput2 = new FairParAsciiFileIo();
151  parInput2->open(par2Str.Data(),"in");
152  rtdb->setSecondInput(parInput2);
153  }
154 
155  // ------------------------------------------------------------------------
156 
157  FairTask* fairTaskList = dynamic_cast<FairTask*>(fInput->FindObject("FairTaskList"));
158 
159  if ( fairTaskList != 0 ) {
160  LOG(info) << "FairAnaSelector::Init(): FairTask = \"" << fairTaskList << "\"";
161  fRunAna->SetTask(fairTaskList);
162  }
163 
164  LOG(info) << "FairAnaSelector::Init(): vvvvv fRunAna->Init() vvvvv";

166  LOG(info) << "FairAnaSelector::Init(): ^^^^^ fRunAna->Init() ^^^^^";

225 Bool_t FairAnaSelector::Process(Long64_t entry)
226 {
227  // The Process() function is called for each entry in the tree (or possibly
228  // keyed object in the case of PROOF) to be processed. The entry argument
229  // specifies which entry in the currently loaded tree is to be processed.
230  // It can be passed to either FairAnaSelector::GetEntry() or TBranch::GetEntry()
231  // to read either all or the required parts of the data. When processing
232  // keyed objects with PROOF, the object is already loaded and is available
233  // via the fObject pointer.
234  //
235  // This function should contain the "body" of the analysis. It can contain
236  // simple or elaborate selection criteria, run algorithms on the data
237  // of the event and typically fill histograms.
238  //
239  // The processing can be stopped by calling Abort().
240  //
241  // Use fStatus to set the return value of TTree::Process().
242  //
243  // The return value is currently not used.
244  // LOG(info) << "FairAnaSelector::Process(): Proceeding to analyze event " << entry << ".";
245 
246  fRunAna->RunOneEvent(entry);
247 
248  // LOG(info) << "FairAnaSelector::Process(): Event " << entry << " analyzed.";
249  return kTRUE;
250 }
