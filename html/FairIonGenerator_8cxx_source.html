47 FairIonGenerator::FairIonGenerator(const Char_t* ionName, Int_t mult,
48  Double_t px, Double_t py, Double_t pz,
49  Double_t vx, Double_t vy, Double_t vz)
50  :FairGenerator(),
51  fMult(mult),
52  fPx(px), fPy(py), fPz(pz),
53  fVx(vx), fVy(vy), fVz(vz),
54  fIon(NULL), fQ(0)
55 
56 {
57 
58  FairRunSim* fRun=FairRunSim::Instance();
59  TObjArray* UserIons=fRun->GetUserDefIons();
60  TObjArray* UserParticles=fRun->GetUserDefParticles();
61  FairParticle* part=0;
62  fIon =static_cast<FairIon*>(UserIons->FindObject(ionName));
63  if(fIon) {
64  fgNIon++;
65  fMult = mult;
66  fPx = Double_t(fIon->GetA()) * px;
67  fPy = Double_t(fIon->GetA()) * py;
68  fPz = Double_t(fIon->GetA()) * pz;
69  fVx = vx;
70  fVy = vy;
71  fVz = vz;
72 
73  } else {
74  part= static_cast<FairParticle*>(UserParticles->FindObject(ionName));
75  if(part) {
76  fgNIon++;
77  TParticle* particle=part->GetParticle();
78  fMult = mult;
79  fPx = Double_t(particle->GetMass()/0.92827231) * px;
80  fPy = Double_t(particle->GetMass()/0.92827231) * py;
81  fPz = Double_t(particle->GetMass()/0.92827231) * pz;
82  fVx = vx;
83  fVy = vy;
84  fVz = vz;
85  }
86  }
87  if(fIon==0 && part==0 ) {
88  LOG(fatal) << "Ion or Particle is not defined !";
89  }
90 
91 }
