129 void Monitor::MonitorHeartbeats()
130 {
131  try
132  {
133  bipc::message_queue mq(bipc::open_or_create, fControlQueueName.c_str(), 1000, 256);
134 
135  unsigned int priority;
136  bipc::message_queue::size_type recvdSize;
137  char msg[256] = {0};
138 
139  while (!fTerminating)
140  {
141  bpt::ptime rcvTill = bpt::microsec_clock::universal_time() + bpt::milliseconds(100);
142  if (mq.timed_receive(&msg, sizeof(msg), recvdSize, priority, rcvTill))
143  {
144  fHeartbeatTriggered = true;
145  fLastHeartbeat = chrono::high_resolution_clock::now();
146  string deviceId(msg, recvdSize);
147  fDeviceHeartbeats[deviceId] = fLastHeartbeat;
148  }
149  else
150  {
151  // cout << "control queue timeout" << endl;
152  }
153  }
154  }
155  catch (bipc::interprocess_exception& ie)
156  {
157  cout << ie.what() << endl;
158  }
159 
160  RemoveQueue(fControlQueueName);
161 }

163 void Monitor::Interactive()
164 {
165  char c;
166  pollfd cinfd[1];
167  cinfd[0].fd = fileno(stdin);
168  cinfd[0].events = POLLIN;
169 
170  struct termios t;
171  tcgetattr(STDIN_FILENO, &t); // get the current terminal I/O structure
172  t.c_lflag &= ~ICANON; // disable canonical input
173  tcsetattr(STDIN_FILENO, TCSANOW, &t); // apply the new settings
174 
175  cout << endl;
176  PrintHelp();
177  cout << endl;
178  PrintHeader();
179 
180  while (!fTerminating)
181  {
182  if (poll(cinfd, 1, 100))
183  {
184  if (fTerminating || gSignalStatus != 0)
185  {
186  break;
187  }
188 
189  c = getchar();
190 
191  switch (c)
192  {
193  case 'q':
194  cout << "\n[q] --> quitting." << endl;
195  fTerminating = true;
196  break;
197  case 'p':
198  cout << "\n[p] --> active queues:" << endl;
199  PrintQueues();
200  break;
201  case 'x':
202  cout << "\n[x] --> closing shared memory:" << endl;
203  Cleanup(fSessionName);
204  break;
205  case 'h':
206  cout << "\n[h] --> help:" << endl << endl;
207  PrintHelp();
208  cout << endl;
209  break;
210  case '\n':
211  cout << "\n[\\n] --> invalid input." << endl;
212  break;
213  default:
214  cout << "\n[" << c << "] --> invalid input." << endl;
215  break;
216  }
217 
218  if (fTerminating)
219  {
220  break;
221  }
222 
223  PrintHeader();
224  }
225 
226  if (fTerminating)
227  {
228  break;
229  }
230 
231  CheckSegment();
232 
233  if (!fTerminating)
234  {
235  cout << "\r";
236  }
237  }
238 
239  tcgetattr(STDIN_FILENO, &t); // get the current terminal I/O structure
240  t.c_lflag |= ICANON; // re-enable canonical input
241  tcsetattr(STDIN_FILENO, TCSANOW, &t); // apply the new settings
242 }

361 void Monitor::Cleanup(const string& sessionName)
362 {
363  string managementSegmentName("fmq_shm_" + sessionName + "_management");
364  try
365  {
366  bipc::managed_shared_memory managementSegment(bipc::open_only, managementSegmentName.c_str());
367  RegionCounter* rc = managementSegment.find<RegionCounter>(bipc::unique_instance).first;
368  if (rc)
369  {
370  cout << "Region counter found: " << rc->fCount << endl;
371  unsigned int regionCount = rc->fCount;
372  for (unsigned int i = 1; i <= regionCount; ++i)
373  {
374  RemoveObject("fmq_shm_" + sessionName + "_region_" + to_string(i));
375  RemoveQueue(string("fmq_shm_" + sessionName + "_region_queue_" + to_string(i)));
376  }
377  }
378  else
379  {
380  cout << "shmem: no region counter found. no regions to cleanup." << endl;
381  }
382 
383  RemoveObject(managementSegmentName.c_str());
384  }
385  catch (bipc::interprocess_exception& ie)
386  {
387  cout << "Did not find '" << managementSegmentName << "' shared memory segment. No regions to cleanup." << endl;
388  }
389 
390  RemoveObject("fmq_shm_" + sessionName + "_main");
391 
392  boost::interprocess::named_mutex::remove(string("fmq_shm_" + sessionName + "_mutex").c_str());
393 
394  cout << endl;
395 }
