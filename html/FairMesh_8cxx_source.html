69 void FairMesh::calculate()
70 {
71 // compute numbers
72 
73  TString tid = fhname + "TID";
74  fMeshTid = new TH2D(tid.Data(),"TID", NXbin, fXmin,fXmax,NYbin,fYmin,fYmax);
75  fMeshTid->Sumw2();
76 
77  TString flu = fhname + "FLU";
78  fMeshFlu = new TH2D(flu.Data(),"Fluence", NXbin, fXmin,fXmax,NYbin,fYmin,fYmax);
79  fMeshFlu->Sumw2();
80 
81  TString seu = fhname + "SEU";
82  fMeshSEU = new TH2D(seu.Data(),"SEU", NXbin, fXmin,fXmax,NYbin,fYmin,fYmax);
83  fMeshSEU->Sumw2();
84 
85 
86  // reseting
87  fMeshTid->Reset();
88  fMeshFlu->Reset();
89  fMeshSEU->Reset();
90 
91 
92  fBinVolume = ((fXmax-fXmin)/NXbin) *
93  ((fYmax-fYmin)/NYbin) *
94  ((fZmax-fZmin)/NZbin);
95 
96  fDiag = TMath::Sqrt(
97  ((fXmax-fXmin)/NXbin)*((fXmax-fXmin)/NXbin) +
98  ((fYmax-fYmin)/NYbin)*((fYmax-fYmin)/NYbin) +
99  ((fZmax-fZmin)/NZbin)*((fZmax-fZmin)/NZbin) );
100 }
