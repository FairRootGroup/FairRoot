205 Bool_t FairMixedSource::Init()
206 {
207  fOutHeader = new FairEventHeader();
208  LOG(info) << "fSBRatiobyN = " << (fSBRatiobyN?"true":"false") << " / fSBRatiobyT = " << (fSBRatiobyT?"true":"false");
209 
210  // LOG(info) << "*********** CHAIN HAS " << fBackgroundChain->GetEntries() << " entries";
211  if(IsInitialized){
212  LOG(info) << "FairMixedSource already initialized";
213  return kTRUE;
214  }
215  if (!fBackgroundChain ) {
216  fBackgroundChain = new TChain(FairRootManager::GetTreeName(), Form("/%s", FairRootManager::GetFolderName()));
217  LOG(info) << "FairMixedSource::Init() chain created";
218  }
219 
220  fBackgroundChain->Add( fRootFile->GetName() );
221  LOG(info) << "*********** CHAIN HAS " << fBackgroundChain->GetEntries() << " entries";
222 
223 // Get the folder structure from file which describes the input tree.
224  // There are two different names possible, so check both.
225  fCbmroot= dynamic_cast <TFolder*> (fRootFile->Get(FairRootManager::GetFolderName()));
226  if(!fCbmroot) {
227  fCbmroot= dynamic_cast <TFolder*> (fRootFile->Get("cbmroot"));
228  if(!fCbmroot) {
229  fCbmroot= dynamic_cast <TFolder*> (fRootFile->Get("cbmout"));
230  if(!fCbmroot) {
231  fCbmroot= gROOT->GetRootFolder()->AddFolder(FairRootManager::GetFolderName(), "Main Folder");
232  } else {
233  fCbmroot->SetName(FairRootManager::GetFolderName());
234  }
235  }
236  }
237  // Get The list of branches from the input file and add it to the
238  // actual list of existing branches.
239  // Add this list of branches also to the map of input trees, which
240  // stores the information which branches belong to which input tree.
241  // There is at least one primary input tree, but there can be many
242  // additional friend trees.
243  // This information is needed to add new files to the correct friend
244  // tree. With this information it is also possible to check if the
245  // input files which are added to the input chain all have the same
246  // branch structure. Without this check it is possible to add trees
247  // with a different branch structure but the same tree name. ROOT
248  // probably only checks if the name of the tree is the same.
249 
250  TList* list= dynamic_cast <TList*> (fRootFile->Get("BranchList"));
251  if(list==0) LOG(fatal) << "No Branch list in input file";
252  TString chainName = fInputTitle;
253  TString ObjName;
254  fInputLevel.push_back(chainName);
255  fCheckInputBranches[chainName] = new std::list<TString>;
256  if(list) {
257  TObjString* Obj=0;
258  LOG(info) << "Enteries in the list " << list->GetEntries();
259  for(Int_t i =0; i< list->GetEntries(); i++) {
260  Obj=dynamic_cast <TObjString*> (list->At(i));
261  if(Obj!=0){
262  ObjName=Obj->GetString();
263  LOG(info) << "Branch name " << ObjName.Data();
264  fCheckInputBranches[chainName]->push_back(ObjName.Data());
265 
266  FairRootManager::Instance()->AddBranchToList(ObjName.Data());
267  }
268  }
269  }
270 
271  gROOT->GetListOfBrowsables()->Add(fCbmroot);
272  fListFolder->Add( fCbmroot );
273 
274  // Store the information about the unique runids in the input file
275  // together with the filename and the number of events for each runid
276  // this information is needed later to check if inconsitencies exist
277  // between the main input chain and any of the friend chains.
278 
279  // GetRunIdInfo(fInFile->GetName(), chainName);
280 
281  // Add all additional input files to the input chain and do a
282  // consitency check
283  std::list<TString>::const_iterator iter;
284  for(iter = fInputChainList.begin(); iter != fInputChainList.end(); iter++) {
285  // Store global gFile pointer for safety reasons.
286  // Set gFile to old value at the end of the routine.R
287  TFile* temp = gFile;
288 
289  // Temporarily open the input file to extract information which
290  // is needed to bring the friend trees in the correct order
291  TFile* inputFile = TFile::Open((*iter));
292  if (inputFile->IsZombie()) {
293  LOG(fatal) << "Error opening the file " << (*iter).Data() << " which should be added to the input chain or as friend chain";
294  }
295 
296  // Check if the branchlist is the same as for the first input file.
297  Bool_t isOk = CompareBranchList(inputFile, chainName);
298  if ( !isOk ) {
299  LOG(fatal) << "Branch structure of the input file " << fRootFile->GetName() << " and the file to be added " << (*iter).Data();
300  return kFALSE;
301  }
302 
303  // Add the runid information for all files in the chain.
304  //GetRunIdInfo(inputFile->GetName(), chainName);
305  // Add the file to the input chain
306  fBackgroundChain->Add( (*iter) );
307 
308  // Close the temporarly file and restore the gFile pointer.
309  inputFile->Close();
310  gFile = temp;
311 
312  }
313  fNoOfEntries = fBackgroundChain->GetEntries();
314  FairRootManager::Instance()->SetInChain(fBackgroundChain,0);
315 
316  for(Int_t i=0; i<fListFolder->GetEntriesFast(); i++) {
317  TFolder* fold = static_cast<TFolder*>(fListFolder->At(i));
318  fEvtHeader = static_cast<FairEventHeader*>(fold->FindObjectAny("EventHeader."));
319  fMCHeader = static_cast<FairMCEventHeader*>(fold->FindObjectAny("MCEventHeader."));
320  if ( fEvtHeader ) {
321  ActivateObject(reinterpret_cast<TObject**>(&fEvtHeader),"EventHeader.");
322  }
323  if ( fMCHeader ) {
324  ActivateObject(reinterpret_cast<TObject**>(&fMCHeader),"MCEventHeader.");
325  }
326  }
327  FairRootManager::Instance()->SetListOfFolders(fListFolder);
328 
329  fBackgroundChain->GetEntry(0);
330  if ( fEvtHeader ) {
331  fOutHeader->SetRunId (fEvtHeader->GetRunId());
332  }
333  if ( fMCHeader ) {
334  fOutHeader->SetRunId (fMCHeader->GetRunID());
335  }
336 
337 
338  LOG(info) << "Entries in this Source " << fNoOfEntries << " ------------";
339  return kTRUE;

343 Int_t FairMixedSource::ReadEvent(UInt_t i)
344 {
345  SetEventTime();
346 
347  Double_t SBratio=gRandom->Uniform(0,1);
348  Bool_t GetASignal=kFALSE;
349 
350  if(fSBRatiobyN || fSBRatiobyT ) {
351  std::map<UInt_t, Double_t>::const_iterator iterN;
352  Double_t ratio=0;
353  if(fCurrentEntryNo==0) {
354  for(iterN = fSignalBGN.begin(); iterN != fSignalBGN.end(); iterN++) {
355  ratio+=iterN->second;
356  fSignalBGN[iterN->first]=ratio;
357  LOG(debug) << "--------------Set signal no. " << iterN->first << " weight " << ratio << ".";
358  }
359  }
360  ratio=0;
361  for(iterN = fSignalBGN.begin(); iterN != fSignalBGN.end(); iterN++) {
362  ratio=iterN->second;
363  LOG(debug) << "---Check signal no. " << iterN->first << " SBratio " << SBratio << " : ratio " << ratio;
364  if(SBratio <=ratio) {
365  TChain* chain = fSignalTypeList[iterN->first];
366  UInt_t entry = fCurrentEntry[iterN->first];
367  chain->GetEntry(entry);
368  fOutHeader->SetMCEntryNumber(entry);
369  fOutHeader->SetInputFileId(iterN->first);
370  fOutHeader->SetEventTime(GetEventTime());
371  GetASignal=kTRUE;
372  fCurrentEntry[iterN->first]=entry+1;
373  LOG(debug) << "---Get entry No. " << entry << " from signal chain number --- " << iterN->first << " --- ";
374  break;
375  }
376  }
377  if(!GetASignal) {
378  UInt_t entry = fCurrentEntry[0];
379  fBackgroundChain->GetEntry(entry);
380  fOutHeader->SetMCEntryNumber(entry);
381  fOutHeader->SetInputFileId(0); //Background files has always 0 as Id
382  fOutHeader->SetEventTime(GetEventTime());
383  fCurrentEntry[0]=entry+1;
384  LOG(debug) << "---Get entry from background chain --- ";
385  }
386 
387  }
388 
389  fCurrentEntryNo=i;
390  fOutHeader->SetEventTime(GetEventTime());
391  LOG(debug) << "--Event number --- " << fCurrentEntryNo << " with time ----" << GetEventTime();
392 
393  return 0;

415 Bool_t FairMixedSource::CompareBranchList(TFile* fileHandle, TString inputLevel)
416 {
417  // fill a set with the original branch structure
418  // This allows to use functions find and erase
419  std::set<TString> branches;
420  std::list<TString>::const_iterator iter;
421  for(iter = fCheckInputBranches[inputLevel]->begin();
422  iter != fCheckInputBranches[inputLevel]->end(); iter++) {
423  branches.insert(*iter);
424  }
425 
426  // To do so we have to loop over the branches in the file and to compare
427  // the branches in the file with the information stored in
428  // fCheckInputBranches["InputChain"]. If both lists are equal everything
429  // is okay
430 
431  // Get The list of branches from the input file one by one and compare
432  // it to the reference list of branches which is defined for this tree.
433  // If a branch with the same name is found, this branch is removed from
434  // the list. If in the end no branch is left in the list everything is
435  // fine.
436  set<TString>::iterator iter1;
437  TList* list= dynamic_cast <TList*> (fileHandle->Get("BranchList"));
438  if(list) {
439  TObjString* Obj=0;
440  for(Int_t i =0; i< list->GetEntries(); i++) {
441  Obj=dynamic_cast <TObjString*> (list->At(i));
442  iter1=branches.find(Obj->GetString().Data());
443  if (iter1 != branches.end() ) {
444  branches.erase (iter1);
445  } else {
446  // Not found is an error because branch structure is
447  // different. It is impossible to add to tree with a
448  // different branch structure
449  return kFALSE;
450  }
451  }
452  }
453  // If the size of branches is !=0 after removing all branches also in the
454  // reference list, this is also a sign that both branch list are not the
455  // same
456  if (branches.size() != 0 ) {
457  return kFALSE;
458  }
459 
460  return kTRUE;
461 }
