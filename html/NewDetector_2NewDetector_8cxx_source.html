77 Bool_t NewDetector::ProcessHits(FairVolume* vol)
78 {
81  //Set parameters at entrance of volume. Reset ELoss.
82  if ( gMC->IsTrackEntering() ) {
83  fELoss = 0.;
84  fTime = gMC->TrackTime() * 1.0e09;
85  fLength = gMC->TrackLength();
86  gMC->TrackPosition(fPos);
87  gMC->TrackMomentum(fMom);
88  }
89 
90  // Sum energy loss for all steps in the active volume
91  fELoss += gMC->Edep();
92 
93  // Create NewDetectorPoint at exit of active volume
94  if ( gMC->IsTrackExiting() ||
95  gMC->IsTrackStop() ||
96  gMC->IsTrackDisappeared() ) {
97  fTrackID = gMC->GetStack()->GetCurrentTrackNumber();
98  fVolumeID = vol->getMCid();
99  if (fELoss == 0. ) { return kFALSE; }
100  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
101  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
102  fELoss);
103 
104  // Increment number of NewDetector det points in TParticle
105  FairStack* stack = (FairStack*) gMC->GetStack();
106  stack->AddPoint(kNewDetector);
107  }
108 
109  return kTRUE;
110 }

147 void NewDetector::ConstructGeometry()
148 {
153  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
154  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
155  NewDetectorGeo* Geo = new NewDetectorGeo();
156  Geo->setGeomFile(GetGeometryFileName());
157  geoFace->addGeoModule(Geo);
158 
159  Bool_t rc = geoFace->readSet(Geo);
160  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
161  TList* volList = Geo->getListOfVolumes();
162 
163  // store geo parameter
164  FairRun* fRun = FairRun::Instance();
165  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
166  NewDetectorGeoPar* par=(NewDetectorGeoPar*)(rtdb->getContainer("NewDetectorGeoPar"));
167  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
168  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
169 
170  TListIter iter(volList);
171  FairGeoNode* node = NULL;
172  FairGeoVolume* aVol=NULL;
173 
174  while( (node = (FairGeoNode*)iter.Next()) ) {
175  aVol = dynamic_cast<FairGeoVolume*> ( node );
176  if ( node->isSensitive() ) {
177  fSensNodes->AddLast( aVol );
178  } else {
179  fPassNodes->AddLast( aVol );
180  }
181  }
182  par->setChanged();
183  par->setInputVersion(fRun->GetRunId(),1);
184 
185  ProcessNodes ( volList );
186 }
