111  Int_t Np=tr->GetNpoints();
112  fTrList= GetTrGroup(P);

172 TEveTrackList* FairMCTracks::GetTrGroup(TParticle* P)
173 {
174 
175  fTrList=0;
176  for (Int_t i=0; i<fEveTrList->GetEntriesFast(); i++) {
177  TEveTrackList* TrListIn=static_cast<TEveTrackList*>(fEveTrList->At(i));
178  if ( strcmp(TrListIn->GetName(),P->GetName())==0 ) {
179  fTrList= TrListIn;
180  break;
181  }
182  }
183  if(fTrList ==0) {
184  fTrPr=new TEveTrackPropagator();
185  fTrList= new TEveTrackList(P->GetName(),fTrPr );
186  fTrList->SetMainColor(fEventManager->Color(P->GetPdgCode()));
187  fEveTrList->Add(fTrList);
188  gEve->AddElement( fTrList ,fEventManager );
189  fTrList->SetRnrLine(kTRUE);
190  }
191  return fTrList;
