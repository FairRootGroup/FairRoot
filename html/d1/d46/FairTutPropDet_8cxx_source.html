66 Bool_t FairTutPropDet::ProcessHits(FairVolume* vol)
67 {
68 
70  // Set parameters at entrance of volume. Reset ELoss.
71  if (TVirtualMC::GetMC()->IsTrackEntering()) {
72  fELoss = 0.;
73  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
74  fLength = TVirtualMC::GetMC()->TrackLength();
75  TVirtualMC::GetMC()->TrackPosition(fPos);
76  TVirtualMC::GetMC()->TrackMomentum(fMom);
77  }
78 
79  // Sum energy loss for all steps in the active volume
80  fELoss += TVirtualMC::GetMC()->Edep();
81 
82  // Create FairTutPropPoint at exit of active volume
83  if (TVirtualMC::GetMC()->IsTrackExiting() || TVirtualMC::GetMC()->IsTrackStop()
84  || TVirtualMC::GetMC()->IsTrackDisappeared()) {
85  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
86  fVolumeID = vol->getMCid();
87 
88  if (fELoss == 0.) {
89  return kFALSE;
90  }
91 
92  // Taking stationNr and sectorNr from string is almost effortless.
93  // Simulation of 100k events with 5 pions without magnetic field takes:
94  // - Real time 142.366 s, CPU time 140.32s WITH USING VolPath TO GET fVolumeID
95  // - Real time 142.407 s, CPU time 140.64s WITHOUT THE FOLLOWING TString OPERATIONS
96  {
97  TString detPath = TVirtualMC::GetMC()->CurrentVolPath();
98  detPath.Remove(0, detPath.Last('/') + 1);
99  detPath.Remove(0, detPath.First("Pixel") + 5);
100  Int_t stationNr = detPath.Atoi();
101  detPath.Remove(0, detPath.First("_") + 1);
102  Int_t sectorNr = detPath.Atoi();
103  fVolumeID = stationNr * 256 + sectorNr;
104  }
105 
106  AddHit(fTrackID,
107  fVolumeID,
108  TVector3(fPos.X(), fPos.Y(), fPos.Z()),
109  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()),
110  fTime,
111  fLength,
112  fELoss);
113 
114  // Increment number of FairTutPropDet points in TParticle
115  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
116  stack->AddPoint(kTutProp);
117  }
118 
119  return kTRUE;
120 }
