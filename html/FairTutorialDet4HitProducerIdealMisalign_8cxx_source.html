138 void FairTutorialDet4HitProducerIdealMisalign::Exec(Option_t* /*opt*/)
139 {
140 
141  fHitArray->Clear();
142 
143  // Declare some variables
144  FairTutorialDet4Point* point = NULL;
145  Int_t detID = 0; // Detector ID
146 // Int_t trackID = 0; // Track index
147  Double_t x, y, z; // Position
148  Double_t dx = 0.1; // Position error
149 // Double_t tof = 0.; // Time of flight
150  TVector3 pos, dpos; // Position and error vectors
151 
152  // Loop over TofPoints
153  Int_t nHits = 0;
154  Int_t nPoints = fPointArray->GetEntriesFast();
155  for (Int_t iPoint=0; iPoint<nPoints; iPoint++) {
156  point = static_cast<FairTutorialDet4Point*>(fPointArray->At(iPoint));
157  if ( ! point) { continue; }
158 
159  // Detector ID
160  detID = point->GetDetectorID();
161 
162  // MCTrack ID
163 // trackID = point->GetTrackID();
164 
165  if(fDoMisalignment) {
166 
167  Float_t cosAlpha = TMath::Cos(fRotZ.At(detID));
168  Float_t sinAlpha = TMath::Sin(fRotZ.At(detID));
169 
170  // Determine hit position
171  x = (point->GetX()*cosAlpha + point->GetY()*sinAlpha)-fShiftX.At(detID);
172  y = (-point->GetX()*sinAlpha + point->GetY()*cosAlpha)-fShiftY.At(detID);
173  z = point->GetZ();
174 
175  LOG(debug)<<"Pos before misalignment: "<< point->GetX() <<", "
176  << point->GetY() <<", "<< point->GetZ();
177  LOG(debug)<<"Pos after misalignment: "<< x <<", "
178  << y <<", "<< z;
179 
180  x = x + GetHitErr(0.1);
181  y = y + GetHitErr(0.1);
182 
183  LOG(debug2)<<"Missallign hit by "<<fShiftX.At(detID)<<" cm in x- and "
184  << fShiftY.At(detID)<<" cm in y-direction.";
185 
186  // Time of flight
187 // tof = point->GetTime();
188 
189  // Create new hit
190  pos.SetXYZ(x,y,z);
191  dpos.SetXYZ(dx, dx, 0.);
192  new ((*fHitArray)[nHits]) FairTutorialDet4Hit(detID, iPoint, pos, dpos);
193  nHits++;
194  } else {
195 
196  // Determine hit position
197  x = point->GetX();
198  y = point->GetY();
199  z = point->GetZ();
200 
201  LOG(info)<<"Position: "<<x<<", "<<y<<", "<<z;
202 
203  Double_t local[3] = {x, y, z};
204  Double_t global[3];
205 
206  fGeoHandler->LocalToGlobal(local, global, detID);
207 
208  x = global[0] + GetHitErr(0.1);
209  y = global[1] + GetHitErr(0.1);
210  z = global[2];
211 
212  LOG(info)<<"Position: "<<x<<", "<<y<<", "<<z;
213  LOG(info)<<"****";
214  // Time of flight
215  //tof = point->GetTime();
216 
217  // Create new hit
218  pos.SetXYZ(x,y,z);
219  dpos.SetXYZ(dx, dx, 0.);
220  new ((*fHitArray)[nHits]) FairTutorialDet4Hit(detID, iPoint, pos, dpos);
221  nHits++;
222 
223  }
224  }
225  // Event summary
226  LOG(debug)<< "Create " << nHits << " TutorialDetHits out of "
227  << nPoints << " TutorilaDetPoints created.";
228

232 Double_t FairTutorialDet4HitProducerIdealMisalign::GetHitErr(Double_t sigma)
233 {
234  Double_t err = gRandom->Gaus(0, sigma);
235  return (TMath::Abs(err) < 3 * sigma) ? err : (err > 0) ? 3 * sigma : -3 * sigma;
236 }
