14 // FairGeoSphe
15 //
16 // class for the GEANT shape SPHE
17 //
18 // The size of a SPHE is defined by 3 'points' (The z-components are not used)
19 // point 0: inner radius of the shell
20 // outer radius of the shell;
21 // point 1: starting polar angle of the shell,
22 // ending polar angle of the shell;
23 // point 2: starting azimuthal angle of the shell,
24 // ending azimuthal angle of the shell; )
25 //
26 // The intrinsic coordinate system of a SPHE, which sits in the CAVE and is
27 // not rotated, is identical with the laboratory system.
28 //

68 Int_t FairGeoSphe::readPoints(std::fstream* pFile,FairGeoVolume* volu)
69 {
70  // reads the 3 'points' decribed above from ascii file
71  // if the array of points is not existing in the volume it is created and
72  // the values are stored inside
73  // returns the number of points
74  if (!pFile) { return 0; }
75  if (volu->getNumPoints()!=nPoints) { volu->createPoints(nPoints); }
76  Double_t x,y;
77  const Int_t maxbuf=155;
78  Text_t buf[maxbuf];
79  for(Int_t i=0; i<nPoints; i++) {
80  pFile->getline(buf,maxbuf);
81  sscanf(buf,"%lf%lf",&x,&y);
82  volu->setPoint(i,x,y,0.0);
83  }
84  return nPoints;
85 }

115 TArrayD* FairGeoSphe::calcVoluParam(FairGeoVolume* volu)
116 {
117  // calculates the parameters needed to create the shape SPHE
118  Double_t fac=10.;
119  FairGeoVector& v0=*(volu->getPoint(0));
120  FairGeoVector& v1=*(volu->getPoint(1));
121  FairGeoVector& v2=*(volu->getPoint(2));
122  param->AddAt(v0(0)/fac,0);
123  param->AddAt(v0(1)/fac,1);
124  param->AddAt(v1(0),2);
125  param->AddAt(v1(1),3);
126  param->AddAt(v2(0),4);
127  param->AddAt(v2(1),5);
128  return param;
129 }
