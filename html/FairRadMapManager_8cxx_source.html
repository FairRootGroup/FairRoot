93  cout << "RadMapMan initialized" << endl;

95  // compute once the masses of the volumes in this simulation and store them in a TMap object
96 
97  Int_t volumeiterator=0,lastvolume=0;
98  Double_t vmass;
99  TObjArray* volumelist;
100  TGeoVolume* myvolume;
101  const char* volumename;
102  fMassMap = new TMap(lastvolume+1,0);
103 
104  volumelist = gGeoManager->GetListOfVolumes();
105 
106  lastvolume = volumelist->GetLast();
107 
108  volumeiterator=0;
109 
110  cout << "RadMapMan: calculating the masses for " << lastvolume << " volumes in this simulation" << endl;
111 
112  while ( volumeiterator<=lastvolume ) {
113  volumename = volumelist->At(volumeiterator)->GetName();
114  myvolume = gGeoManager->GetVolume(volumename);
115  vmass = myvolume->WeightA(); // calculate weight analytically
116  TVectorD* vomass= new TVectorD(1,1,vmass,"END"); // 1-dim vector
117 
118  fMassMap->Add(myvolume,vomass);
119 
120  cout << myvolume->GetName() << " has " << vmass << " kg" << endl;
121 
122  volumeiterator++;
123  }

159

162  // cout << actVolume->GetName() << " has " << fActMass << " kg" << endl;
