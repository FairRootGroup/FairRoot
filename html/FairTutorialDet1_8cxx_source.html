95 Bool_t FairTutorialDet1::ProcessHits(FairVolume* vol)
96 {
99  LOG(debug) << "In FairTutorialDet1::ProcessHits";
100  //Set parameters at entrance of volume. Reset ELoss.
101  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {
102  fELoss = 0.;
103  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
104  fLength = TVirtualMC::GetMC()->TrackLength();
105  TVirtualMC::GetMC()->TrackPosition(fPos);
106  TVirtualMC::GetMC()->TrackMomentum(fMom);
107  }
108 
109  // Sum energy loss for all steps in the active volume
110  fELoss += TVirtualMC::GetMC()->Edep();
111 
112  // Create FairTutorialDet1Point at exit of active volume
113  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
114  TVirtualMC::GetMC()->IsTrackStop() ||
115  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
116  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
117  fVolumeID = vol->getMCid();
118  if (fELoss == 0. ) { return kFALSE; }
119  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
120  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
121  fELoss);
122 
123  // Increment number of tutorial det points in TParticle
124  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
125  stack->AddPoint(kTutDet);
126  }
127 
128  return kTRUE;
129 }

202 void FairTutorialDet1::SetSensitiveVolumes()
203 {
204  TList* volList = fgGeo->getListOfVolumes();
205 
206  // store geo parameter
207  FairRun* fRun = FairRun::Instance();
208  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
209  FairTutorialDet1GeoPar* par=static_cast<FairTutorialDet1GeoPar*>(rtdb->getContainer("FairTutorialDet1GeoPar"));
210  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
211  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
212 
213  TListIter iter(volList);
214  FairGeoNode* node = NULL;
215  FairGeoVolume* aVol=NULL;
216 
217  while( (node = static_cast<FairGeoNode*>(iter.Next())) ) {
218  aVol = dynamic_cast<FairGeoVolume*> ( node );
219  if ( node->isSensitive() ) {
220  fSensNodes->AddLast( aVol );
221  } else {
222  fPassNodes->AddLast( aVol );
223  }
224  }
225  par->setChanged();
226  par->setInputVersion(fRun->GetRunId(),1);
227 
228  ProcessNodes ( volList );
229 }
