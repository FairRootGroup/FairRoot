819 void FairGeanePro::Track2ToLine( TVector3 X1, TVector3 X2, TVector3 w1,
820  TVector3 w2, TVector3& Pfinal, TVector3& Pwire,
821  int& Iflag, double& Dist, double& Length)
822 {
823  // Closest approach to a line from 2 GEANE points
824  //
825  // METHOD: the nearest points on the two lines
826  // x1,x2 and w1,w2 is found.
827  // The method is described in:
828  // http://softsurfer.com/Archive/algorithm_0106/algorithm_0106.htm
829  // http://www.geometrictools.com/Documentation/DistanceLine3Line3.pdf.
830  //
831  // INPUT: x1, x2 closest appoach GEANE points
832  // w1, w2 points of the line (wire) to approach
833  //
834  // OUTPUT: Pfinal point of closest approach on the track
835  // Pwire point of closest approach on the wire
836  // Dist distance between Pfian and w1
837  // Length arc length to add to the GEANE length of x1
838  // Iflag =1 when Pwire is outside [w1,w2]
839  // In this case, when w1 and w2 are the extremes
840  // of the wire, the user could remake the procedure
841  // by calling Track3ToPoint or Track2ToPoint, where the
842  // Point is w1 or w2;
843  // = 2 when the two lines are parallel and the solution
844  // does not exists.
845  //
846  // Authors: Andrea Fontana and Alberto Rotondi 20 MAy 2007
847  //
848 
849  TVector3 x21, x32, w21;
850  TVector3 xw1, xw2;
851 
852  double a1, b1, c1, d1, e1;
853  double Delta1;
854 
855  double Eps = 1.E-08;
856 
857  Iflag =0;
858 
859  // line-line distance
860  x21 = X2-X1;
861  w21 = w2-w1;
862 
863  xw1 = X1-w1;
864  xw2 = X2-w1;
865 
866  a1 = x21.Mag2();
867  b1 = x21.Dot(w21);
868  c1 = w21.Mag2();
869  d1 = xw1.Dot(x21);
870  e1 = xw1.Dot(w21);
871 
872  Delta1 = a1*c1-b1*b1;
873 
874  if(Delta1 > Eps) {
875  double t1 = (a1*e1-b1*d1)/Delta1;
876  double s1 = (b1*e1-c1*d1)/Delta1;
877 
878  Pfinal = (X1 + x21*s1);
879  Pwire = (w1 + w21*t1);
880  Length = s1*x21.Mag();
881  Dist= (Pfinal-Pwire).Mag();
882 
883  } else {
884  // lines are paralllel, no solution does exist
885  Pfinal.SetXYZ(0.,0.,0.);
886  Pwire.SetXYZ(0.,0.,0.);
887  Dist=0.;
888  Length=0.;
889  Iflag = 2;
890  return;
891  }
892  // flag when the point on the wire is outside (w1,w2)
893  if((((Pwire[0]<w1[0] && Pwire[0]<w2[0]) || (w2[0]<Pwire[0] && w1[0]<Pwire[0]))
894  && (fabs(Pwire[0]-w1[0]) > 1e-11 && fabs(Pwire[0]- w2[0]) > 1e-11))
895  || (((Pwire[1]<w1[1] && Pwire[1]<w2[1]) || (w2[1]<Pwire[1] && w1[1]<Pwire[1]))
896  && (fabs(Pwire[1]-w1[1]) > 1e-11 && fabs(Pwire[1]- w2[1]) > 1e-11))
897  || (((Pwire[2]<w1[2] && Pwire[2]<w2[2]) || (w2[2]<Pwire[2] && w1[2]<Pwire[2]))
898  && (fabs(Pwire[2]-w1[2]) > 1e-11 && fabs(Pwire[2]- w2[2]) > 1e-11))) {
899  Iflag=1;
900  }
901 }

903 void FairGeanePro::Track2ToPoint( TVector3 X1, TVector3 X2, TVector3 w1, TVector3& Pfinal,
904  double& Dist, double& Length, int& quitFlag)
905 {
906  //
907  // Closest approach to a point from 2 GEANE points
908  //
909  // METHOD: the nearest point to w1 on the line x1,x2
910  // is found. Elementary vector calculus is used!
911  //
912  // INPUT: x1, x2 closest appoach GEANE points
913  // w1 point to approach w1
914  //
915  // OUTPUT: Pfinal point of closest approach
916  // Dist distance between Pfinal and w1
917  // Length arc length to add to the GEANE length of x1
918  // quitFlag error flag which will be set to 1 if points dont form line
919  // Authors: Andrea Fontana and Alberto Rotondi May 2007
920  //
921 
922  quitFlag = 0;
923 
924  TVector3 u21;
925 
926  double a, t1;
927 
928  // w-point - x-line distance
929  double d=(X2-X1).Mag();
930  if(fabs(d)<1.E-8) {
931  quitFlag=1;
932  return;
933  }
934  a= 1./d;
935 
936  u21 = (X2-X1).Unit();
937 
938  // output

945 void FairGeanePro::Track3ToLine(TVector3 X1, TVector3 X2, TVector3 X3,
946  TVector3 w1, TVector3 w2,
947  // output
948  TVector3& Pfinal, TVector3& Wire,
949  int& Iflag, double& Dist,
950  double& Length, double& Radius)
951 {
952  // Find the closest approach points between a curve (helix)
953  // and a line (wire)
954  //
955  // METHOD:the classical Eberly method is used: see
956  // http://www.geometrictools.com/Documentation/DistanceLine3Circle.pdf.
957  // (see also www.geometrictools.com for the other formulae used
958  // in this interface).
959  // The 4-degree polynomial resulting for the line parameter t is
960  // solved with the efficient SolveQuartic Root routine.
961  // The minimal distance solution is found by using our Track3ToPoint
962  // routine
963  //
964  // INPUT: x1, x2, x3 closest appoach GEANE points
965  // w1, w2 points of the line (wire) to approach
966  //
967  // OUTPUT: Pfinal track point of closest approach
968  // Wire line point of closest approach
969  // Iflag = 1, the points are on a straight line
970  // within the precision of the method (20 micron).
971  // In this case the user should recall Track2ToPoint
972  // = 2, Pwire is outside [w1,w2]
973  // In this case, when w1 and w2 are the extremes
974  // of the wire, the user could remake the procedure
975  // by calling Track3ToPoint, where the Point is w1 or w2.
976  // = 3 both conditions 1 and 2 are encountered
977  // = 4, the method failed for mathematical reasons.
978  // In this case the user should use Track2ToLine where
979  // x1=x1 and x2=x3.
980  // Dist distance between Pfinal and Wire
981  // Length arc length to add to the GEANE length of x1
982  // Radius radius of the found circle
983  //
984  // Authors: Andrea Fontana and Alberto Rotondi 20 June 2007
985  //
986 
987  TVector3 xp1, xp2, xp3, xp32;
988  TVector3 x21, x31;
989  TVector3 e1, e2, e3, aperp;
990  TVector3 Ppfinal, Pwire;
991  TVector3 wp1, wp2, wpt, xR, xpR;
992  TVector3 xw1, xw2;
993  TVector3 px;
994 
995  double T[3][3], TM1[3][3];
996 
997  TVector3 N, M, D, B, Pointw;
998  double a0, a1, b0, b1, c0, c1, c2;
999  double d0, d1, d2, d3, d4, sol4[4], dmin;
1000  double Angle;
1001  double dx;
1002  int it, imin;
1003 
1004  Iflag = 0;
1005 
1006  // go to the circle plane: matrix of director cosines
1007 
1008  x21 = X2-X1;
1009  e1 = x21.Unit();
1010  T[0][0] = e1.X();
1011  T[0][1] = e1.Y();
1012  T[0][2] = e1.Z();
1013 
1014  x31 = X3-X1;
1015  e3 = e1.Cross(x31);
1016  // if the points are on the same line
1017  if(e3.Mag() < 1e-8) {
1018  Iflag = 1;
1019  return;
1020  }
1021  e3 = e3.Unit();
1022 
1023  T[2][0] = e3.X();
1024  T[2][1] = e3.Y();
1025  T[2][2] = e3.Z();
1026 
1027  e2 = e3.Cross(e1);
1028  T[1][0] = e2.X();
1029  T[1][1] = e2.Y();
1030  T[1][2] = e2.Z();
1031 
1032  // new coordinates
1033  for(int i=0; i<3; i++) {
1034  xp1[i]=0.;
1035  xp2[i]=0.;
1036  xp3[i]=0.;
1037  wp1[i]=0.;
1038  wp2[i]=0.;
1039  }
1040  for(int i=0; i<3; i++) {
1041  xp1[i] = 0.;
1042  for(int j=0; j<3; j++) {
1043  TM1[i][j] = T[j][i];
1044  xp2[i] += T[i][j] * (X2[j]-X1[j]);
1045  xp3[i] += T[i][j] * (X3[j]-X1[j]);
1046  wp1[i] += T[i][j] * (w1[j]-X1[j]);
1047  wp2[i] += T[i][j] * (w2[j]-X1[j]);
1048  }
1049  }
1050 
1051  // radius and center
1052 
1053  xp32= xp3 - xp2;
1054  xpR[0] = 0.5*xp2[0];
1055  if(fabs(xp3[1])<1.E-8) {
1056  Iflag = 4;
1057  return;
1058  }
1059  xpR[1] = 0.5*(xp32[0]*xp3[0]/xp3[1]+ xp3[1]);
1060  xpR[2] = 0.;
1061  Radius = sqrt(pow(xpR[0]-xp1[0],2)+pow(xpR[1]-xp1[1],2));
1062 
1063  // Eberly's method
1064 
1065  B = wp1;
1066  M = wp2 - wp1;
1067  D = wp1-xpR;
1068  N.SetXYZ(0.,0.,1.);
1069 
1070  a0 = M.Dot(D);
1071  a1 = M.Dot(M);
1072  b0 = M.Dot(D) -(N.Dot(M))*(N.Dot(D));
1073  b1 = M.Dot(M) -(N.Dot(M))*(N.Dot(M));
1074  c0 = D.Dot(D) -(N.Dot(D))*(N.Dot(D));
1075  c1 = b0;
1076  c2 = b1;
1077 
1078  d0 = a0*a0*c0 -b0*b0*Radius*Radius;
1079  d1 = 2.*(a0*a1*c0+a0*a0*c1-b0*b1*Radius*Radius);
1080  d2 = a1*a1*c0+4.*a0*a1*c1+a0*a0*c2-b1*b1*Radius*Radius;
1081  d3 = 2.*(a1*a1*c1+a0*a1*c2);
1082  d4 = a1*a1*c2;
