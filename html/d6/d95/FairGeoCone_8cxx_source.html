14 // FairGeoCone
15 //
16 // class for the GEANT shape CONE
17 //
18 // The size of a CONE is defined by 4 'points'.
19 // point 0: origin of starting circle of the cone;
20 // point 1: inner radius of starting circle,
21 // outer radius of starting circle;
22 // (z-component not used)
23 // point 2: origin of ending circle of the cone;
24 // point 3: inner radius of ending circle,
25 // outer radius of ending circle;
26 // (z-component not used)
27 // Warning: The x- and y-values of point 0 and 2 have to be the same!!!!
28 // A rotation has to be desribed via the rotation matrix.
29 //
30 // The intrinsic coordinate system of a CONE, which sits in the CAVE and is
31 // not rotated, is identical with the laboratory system.
32 //

78 Int_t FairGeoCone::readPoints(std::fstream* pFile,FairGeoVolume* volu)
79 {
80  // reads the 4 'points' decribed above from ascii file
81  // if the array of points is not existing in the volume it is created and
82  // the values are stored inside
83  // returns the number of points
84  if (!pFile) { return 0; }
85  if (volu->getNumPoints()!=nPoints) { volu->createPoints(nPoints); }
86  Double_t x,y,z;
87  const Int_t maxbuf=155;
88  Text_t buf[maxbuf];
89  for(Int_t i=0; i<nPoints; i++) {
90  pFile->getline(buf,maxbuf);
91  sscanf(buf,"%lf%lf%lf",&x,&y,&z);
92  volu->setPoint(i,x,y,z);
93  i++;
94  pFile->getline(buf,maxbuf);
95  sscanf(buf,"%lf%lf",&x,&y);
96  volu->setPoint(i,x,y,0.0);
97  }
98  return nPoints;
99 }

101 Bool_t FairGeoCone::writePoints(std::fstream* pFile,FairGeoVolume* volu)
102 {
103  // writes the 4 'points' decribed above to ascii file
104  if (!pFile) { return kFALSE; }
105  Text_t buf[155];
106  for(Int_t i=0; i<nPoints; i++) {
107  FairGeoVector& v=*(volu->getPoint(i));
108  if (i==0 || i==2) { sprintf(buf,"%9.3f%10.3f%10.3f\n",v(0),v(1),v(2)); }
109  else { sprintf(buf,"%9.3f%10.3f\n",v(0),v(1)); }
110  pFile->write(buf,strlen(buf));
111  }
112  return kTRUE;
113 }

125 TArrayD* FairGeoCone::calcVoluParam(FairGeoVolume* volu)
126 {
127  // calculates the parameters needed to create the shape CONE
128  Double_t fac=10.;
129  FairGeoVector v=*(volu->getPoint(2)) - *(volu->getPoint(0));
130  param->AddAt(TMath::Abs(v(2))/fac/2.,0);
131  FairGeoVector& v1=*(volu->getPoint(1));
132  param->AddAt(v1(0)/fac,1);
133  param->AddAt(v1(1)/fac,2);
134  FairGeoVector& v3=*(volu->getPoint(3));
135  param->AddAt(v3(0)/fac,3);
136  param->AddAt(v3(1)/fac,4);
137  return param;
138 }
