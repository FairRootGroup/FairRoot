191  sLmdControl* pLmdControl,

199  if(pLmdControl->iOffsetEntries && (pLmdControl->iOffsetSize == 4)) {
200  elements=pLmdControl->iElements+2;

204  printf("fLmdPutElement: File size exceed\n");

206  }
207  }
208  // save largest size in header

210  pLmdControl->pMbsFileHeader->iMaxWords=pHeader->iWords+4;

214  pLmdControl->pMbsHeader=pHeader;

218  printf("fLmdPutElement: Write error \n");

223  if(pLmdControl->iOffsetEntries) { fLmdOffsetSet(pLmdControl,iReturn/4); }
224  return(LMD__SUCCESS);
225 
226  } // end no buffer

228  iReturn=fLmdWriteBuffer(pLmdControl, (char*)pLmdControl->pBuffer,
229  (pLmdControl->iBufferWords-pLmdControl->iLeftWords)*2);

245  pLmdControl->iLeftWords -= (pHeader->iWords+4);
246  if(pLmdControl->iOffsetEntries) { fLmdOffsetSet(pLmdControl,iReturn); }
247  return(LMD__SUCCESS);

632 uint32_t fLmdGetBuffer(
633  sLmdControl* pLmdControl,
634  sMbsHeader* pMbsHeader,
635  uint32_t iBytes,
636  uint32_t* iElements,
637  uint32_t* iBytesUsed)
638 {
639 
640  sMbsHeader* pm;
641  uint32_t elem=0,leftBytes=0, used, elem_sz;//, *ps, *pd, i,ii, size=0;
642  int32_t iReturn;
643 
644  if(iBytes < pLmdControl->pMbsFileHeader->iMaxWords) {
645  printf("fLmdGetBuffer: %s buffer size %d too small for %d bytes\n",
646  pLmdControl->cFile,iBytes,pLmdControl->pMbsFileHeader->iMaxWords);
647  return(LMD__FAILURE);
648  }
649  if(pMbsHeader == NULL) {
650  printf("fLmdGetBuffer: Need buffer to read\n");
651  return(LMD__FAILURE);
652  }
653  *iBytesUsed=0;
654  *iElements=0;
655  if(pLmdControl->iElements == pLmdControl->pMbsFileHeader->iElements) { return(GETLMD__EOFILE); }
656 
657  // Offset table
658  if(pLmdControl->iOffsetEntries) { // use offsets to read elements fitting in buffer
659  fLmdOffsetElements(pLmdControl,iBytes, &elem, &used);
660  //printf("Read %d bytes of %d, elements %d\n",used,iBytes,elem);
661  iReturn=fLmdReadBuffer(pLmdControl,(char*)pMbsHeader,used);
662  if(iReturn <= 0) {
663  printf("fLmdGetBuffer: EOF: %s\n",pLmdControl->cFile);
664  return(GETLMD__EOFILE);
665  }
666  if(iReturn!=used) {
667  printf("fLmdGetBuffer: LMD read error: unexpected EOF: %s %u %u\n",
668  pLmdControl->cFile, iReturn, used);
669  return(GETLMD__NOLMDFILE);
670  }
671  *iBytesUsed=used;
672  *iElements=elem;
673  if(pLmdControl->iSwap) { fLmdSwap4((uint32_t*)pMbsHeader,iReturn/4); }
674  pLmdControl->iBytes+=iReturn;
675  return(LMD__SUCCESS);
676  }
677  // no offset table
678  // do we have fragment stored?
679  leftBytes = pLmdControl->iLeftWords*2;
680  if(leftBytes>0) {
681  if (leftBytes > iBytes) {
682  printf("fLmdGetBuffer: stored piece of data (%u) larger than provided buffer (%u)\n",
683  leftBytes, iBytes);
684  return(LMD__FAILURE);
685  }
686 
687  if (pLmdControl->pMbsHeader==0) {
688  printf("fLmdGetBuffer: Internal error pMbsHeader==0\n");
689  return(LMD__FAILURE);
690  }
691 
692  memcpy(pMbsHeader, pLmdControl->pMbsHeader, leftBytes);
693  }
694  iReturn = fLmdReadBuffer(pLmdControl,(char*)pMbsHeader+leftBytes, iBytes-leftBytes);
695  if(iReturn <= 0) {
696  printf("fLmdGetBuffer: EOF: %s\n",pLmdControl->cFile);
697  if (leftBytes>0) {
698  printf("fLmdGetBuffer: EOF while we have some rest data (%u)\n", leftBytes);
699  } else {
700  return(GETLMD__EOFILE);
701  }
702  }
703 
704  if(iReturn > (iBytes-leftBytes)) {
705  printf("fLmdGetBuffer: LMD read error %s - too many bytes read %u wants %u",
706  pLmdControl->cFile, iReturn, iBytes-leftBytes);
707  return(GETLMD__NOLMDFILE);
708  }
709 
710  if(pLmdControl->iSwap) { fLmdSwap4((uint32_t*)pMbsHeader+leftBytes/4,iReturn/4); }
711  pLmdControl->iBytes += iReturn;
712  leftBytes += iReturn; // thats what is in the buffer
713  // step through buffer to get number of elements and size
714  pm=pMbsHeader;
715  while(leftBytes >=8) {
716  if(pm->iType == LMD__TYPE_FILE_INDEX_101_2) { break; } // file index is last
717  elem_sz = (pm->iWords+4)*2;
718  if(elem_sz > leftBytes) { break; } // pm valid but incomplete data
719 
720  *iBytesUsed += elem_sz;
721  *iElements += 1;
722  pLmdControl->iElements++;
723  pm = (sMbsHeader*)((char*)pm + elem_sz);
724  leftBytes -= elem_sz;
725  }
726  //printf("Read %d bytes of %d, elements %d\n",*iBytesUsed,iBytes,*iElements);
727  // fragment left? copy to internal buffer
728  if(leftBytes>0) {
729  if(leftBytes > pLmdControl->iBufferWords*2) {
730  printf("fLmdGetBuffer: ERROR: internal buffer overflow. Needed:%d available:%d\n",
731  leftBytes,pLmdControl->iBufferWords*2);
732  return(LMD__FAILURE);
733  } else {
734  memcpy(pLmdControl->pBuffer,pm,leftBytes);
735  }
736  }
737  pLmdControl->iLeftWords = leftBytes/2;
738  if (pLmdControl->iLeftWords>0) {
739  pLmdControl->pMbsHeader = (sMbsHeader*)pLmdControl->pBuffer;
740  } else {
741  pLmdControl->pMbsHeader = 0;
742  }
743 
744  return(LMD__SUCCESS);
745 }

755  if(pLmdControl->pBuffer==NULL) { return(GETLMD__NOBUFFER); } // internal buffer needed
756  if(pLmdControl->pMbsFileHeader->iElements==0) { return(GETLMD__NOMORE); }
757

761

775  else if((pLmdControl->iLeftWords < 4) ||

781 // printf("copy to the begin rest %u bytes", pLmdControl->iLeftWords*2);

788  (pLmdControl->iBufferWords-pLmdControl->iLeftWords)*2);

800

802

806  return (GETLMD__TOOBIG);
807  }
808

810  pM = pLmdControl->pMbsHeader;
811  pLmdControl->pMbsHeader = (sMbsHeader*) ((char*) pM + evsz);
812  pLmdControl->iLeftWords -= evsz/2;
813  pLmdControl->iElements++;
814  *event=pM;
815  return(LMD__SUCCESS);
816  }

819  if(iEvent >= pLmdControl->iOffsetEntries) { return(GETLMD__OUTOF_RANGE); }

823  return(GETLMD__EOFILE);
824  }

826  iReturn=fLmdReadBuffer(pLmdControl,(char*)pLmdControl->pBuffer,i*4);
827  if(iReturn <= 0) {printf("fLmdGetElement: EOF\n"); return(GETLMD__EOFILE);}
828  if(iReturn!=(i*4)) {
829  printf("fLmdGetBuffer: LMD read error: unexpected EOF: %s\n",pLmdControl->cFile);
830  return(GETLMD__EOFILE);
831  }
832  if(pLmdControl->iSwap) { fLmdSwap4((uint32_t*)pLmdControl->pBuffer,iReturn/4); }
833  pLmdControl->pMbsHeader=(sMbsHeader*)pLmdControl->pBuffer;
834  if((pLmdControl->pMbsHeader->iWords+4) != i*2) {
835  printf("fLmdGetElement: Error Event %d: size from table is %d, header %d\n",
836  iEvent,i/2,pLmdControl->pMbsHeader->iWords+4);
837  return(GETLMD__SIZE_ERROR);
838  }
839  pLmdControl->iBytes+=iReturn;
840  *event=pLmdControl->pMbsHeader;
841  return(LMD__SUCCESS);
842  } else { return(GETLMD__NOMORE); }
843  // return zero if no more events
844 }
