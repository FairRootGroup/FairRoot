65 Int_t FairGeoTorus::readPoints(std::fstream* pFile,FairGeoVolume* volu)
66 {
67  // reads nPoints with 3 components from Ascii file
68  // if the array of points is not existing in the volume it is created and
69  // the values are stored inside
70  // returns the number of points
71  if (!pFile) { return 0; }
72  if (volu->getNumPoints()!=nPoints) { volu->createPoints(nPoints); }
73  Double_t x;
74  const Int_t maxbuf=155;
75  Text_t buf[maxbuf];
76  for(Int_t i=0; i<nPoints; i++) {
77  pFile->getline(buf,maxbuf);
78  sscanf(buf,"%lf",&x);
79  volu->setPoint(i,x,0,0);
80  }
81  return nPoints;
82 }
