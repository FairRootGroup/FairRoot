292 double FairRKPropagator::CalculatePointToWireDistance(TVector3 point, TVector3 wire1, TVector3 wire2, TVector3& vwi)
293 {
294  TVector3 ab = wire2 - wire1;
295  TVector3 av = point - wire1;
296 
297  if ( av.Dot(ab) <= 0.0 ) { // Point is lagging behind start of the segment, so perpendicular distance is not viable
298  vwi = wire1;
299  return av.Mag(); // Use distance to start of segment instead
300  }
301 
302  TVector3 bv = point - wire2;
303 
304  if ( bv.Dot(ab) >= 0.0 ) { // Point is advanced past the end of the segment, so perpendicular distance is not viable
305  vwi = wire2;
306  return bv.Mag(); // Use distance to end of the segment instead
307  }
308 
309  vwi = ( ab.Dot(av) / ab.Mag() / ab.Mag() ) * ab;
310  vwi += wire1;
311  return (ab.Cross(av)).Mag() / ab.Mag(); // Perpendicular distance of point to segment
312 }
