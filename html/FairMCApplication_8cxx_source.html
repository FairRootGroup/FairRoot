667 //_____________________________________________________________________________
668 void FairMCApplication::Stepping()
669 {
670 // User actions at each step
671 // ---
672 
673  // Work around for Fluka VMC, which does not call
674  // MCApplication::PreTrack()
675  static Int_t TrackId = 0;
676  if ( fMcVersion ==2 && fMC->GetStack()->GetCurrentTrackNumber() != TrackId ) {
677  PreTrack();
678  TrackId = fMC->GetStack()->GetCurrentTrackNumber();
679  }
680 
681  // Check if the volume with id is in the volume multimap.
682  // If it is not in the map the volume is not a sensitive volume
683  // and we do not call nay of our ProcessHits functions.
684 
685  // If the volume is in the multimap, check in second step if the current
686  // copy is alredy inside the multimap.
687  // If the volume is not in the multimap add the copy of the volume to the
688  // multimap.
689  // In any case call the ProcessHits function for this specific detector.
690  Int_t copyNo;
691  Int_t id = fMC->CurrentVolID(copyNo);
692  Bool_t InMap =kFALSE;
693  fDisVol=0;
694  fDisDet=0;
695  Int_t fCopyNo=0;
696  fVolIter =fVolMap.find(id);
697 
698  if (fVolIter!=fVolMap.end()) {
699 
700  // Call Process hits for FairVolume with this id, copyNo
701  do {
702  fDisVol=fVolIter->second;
703  fCopyNo=fDisVol->getCopyNo();
704  if(copyNo==fCopyNo) {
705  fDisDet=fDisVol->GetDetector();
706  if (fDisDet) {
707  fDisDet->ProcessHits(fDisVol);
708  }
709  InMap=kTRUE;
710  break;
711  }
712  fVolIter++;
713  }
