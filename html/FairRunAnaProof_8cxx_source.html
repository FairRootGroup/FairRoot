321  RunTSBuffers();

347 void FairRunAnaProof::RunOnProof(Int_t NStart,Int_t NStop)
348 {
349  fProofOutputStatus.ToLower();
350  if ( !fProofOutputStatus.Contains("copy") && !fProofOutputStatus.Contains("merge") ) {
351  LOG(warn) << "FairRunAnaProof::RunOnProof. Do not know how to create output \"" << fProofOutputStatus.Data() << "\".";
352  LOG(warn) << "FairRunAnaProof::RunOnProof. Please use SetProofOutputStatus to either \"copy\" or \"merge\".";
353  LOG(warn) << "FairRunAnaProof::RunOnProof. For the current run using the \"merge\" setting.";
354  fProofOutputStatus = "merge";
355  }
356 
357 // TChain* inChain = static_cast<TChain*>(fRootManager->GetInChain());
358  TChain* inChain = static_cast<TChain*>(fProofFileSource->GetInChain());
359  TString par1File = "";
360  TString par2File = "";
361  if ( fRtdb->getFirstInput () ) {
362  par1File = fRtdb->getFirstInput ()->getFilename();
363  }
364  if ( fRtdb->getSecondInput() ) {
365  par2File = fRtdb->getSecondInput()->getFilename();
366  }
367 
368  TString outDir = (fOutputDirectory.Length()>1?fOutputDirectory.Data():gSystem->WorkingDirectory());
369 
370  fProof->AddInput(fTask);
371 
372  fProof->AddInput(new TNamed("FAIRRUNANA_fContainerStatic",(fStatic?"kTRUE":"kFALSE")));
373  fProof->AddInput(new TNamed("FAIRRUNANA_fProofOutputStatus",fProofOutputStatus.Data()));
374  fProof->AddInput(new TNamed("FAIRRUNANA_fOutputDirectory",outDir.Data()));
375  fProof->AddInput(new TNamed("FAIRRUNANA_fOutputFileName",fOutname.Data()));
376  fProof->AddInput(new TNamed("FAIRRUNANA_fParInput1FName",par1File.Data()));
377  fProof->AddInput(new TNamed("FAIRRUNANA_fParInput2FName",par2File.Data()));
378 
379  // uploading packages not needed, as the libraries are in the rootmap now
380  // fProof->ClearPackages();
381  // fProof->UploadPackage(fProofParName.Data());
382  // fProof->EnablePackage(fProofParName.Data());
383 
384  Int_t nofChainEntries = inChain->GetEntries();
385  LOG(info) << "FairRunAnaProof::RunOnProof(): The chain seems to have " << nofChainEntries << " entries.";
386 
387  TObjArray* listOfFiles = inChain->GetListOfFiles();
388  LOG(info) << "FairRunAnaProof::RunOnProof(): There are " << listOfFiles->GetEntries() << " files in the chain.";
389 
390  inChain->SetProof();
391 
392  Int_t nofEventsToAnalyze = NStop-NStart;
393 
394  if ( nofEventsToAnalyze <= 0 ) {
395  LOG(info) << "You requested to analyze events from " << NStart << " to " << NStop << " that is " << nofEventsToAnalyze << " events!!!";
396  nofEventsToAnalyze = nofChainEntries-NStart;
397  LOG(info) << "It will be changed to analyze all events from " << NStart << " to the end of chain (" << nofChainEntries << "), that is to analyze " << nofEventsToAnalyze << " events.";
398  }
399 
400  LOG(info) << "FairRunAnaProof::RunOnProof(): Starting inChain->Process(\"FairAnaSelector\",\"\","
401  << nofEventsToAnalyze << "," << NStart << ")";
402  inChain->Process("FairAnaSelector","",nofEventsToAnalyze,NStart);
403  LOG(info) << "FairRunAnaProof::RunOnProof(): inChain->Process DONE";
404 
405  return;
406 }
