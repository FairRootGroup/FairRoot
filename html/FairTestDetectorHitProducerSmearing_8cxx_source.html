92 
93  fHitsArray->Delete();

105 
106  TVector3 position;

108 
109  TVector3 dposition;

115 // hit = new ((*fHitsArray)[iPoint]) FairTestDetectorHit(point->GetDetectorID(), iPoint, position, dposition);
