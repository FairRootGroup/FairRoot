92 // ----- Accessor to momentum -------------------------------------------
93 void FairTrackParam::Momentum(TVector3& mom) const
94 {
95  Double_t p = ( TMath::Abs(fQp)>1.e-4 ) ?1./TMath::Abs(fQp) :1.e4;
96  Double_t pz = TMath::Sqrt( p*p / ( fTx*fTx + fTy*fTy + 1 ) );
97  Double_t px = fTx * pz;
98  Double_t py = fTy * pz;
99  mom.SetXYZ(px, py, pz);
100 
101 }

131 // ----- Accessor to covariance matrix elements ------------------------
132 Double_t FairTrackParam::GetCovariance(Int_t i, Int_t j) const
133 {
134  if ( i<0 || j<0 || i>4 || j>4 ) {
135  LOG(error) << "FairTrackParam::GetCovariance: Invalid index pair ("
136  << i << "," << j << ") !";
137  return 0;
138  }
139  if (i>j) {
140  Int_t k = i;
141  i = j;
142  j = k;
143  }
144  Int_t index = 0;
145  if (i==0) { index = j; }
146  else if (i==1) { index = 4 + j; }
147  else if (i==2) { index = 7 + j; }
148  else if (i==3) { index = 9 + j; }
149  else if (i==4) { index = 10 + j; }
150  return fCovMatrix[index];
151 }
