132 Bool_t FairParRootFileIo::open(const Text_t* fname, Option_t* option,
133  const Text_t* ftitle, Int_t compress)
134 {
135  // It opens a ROOT file (default option "READ"). An open file will be closed.
136  // The detector I/Os for all detectors defined in the setup are activated.
137 
138  //DB, lock file for NFS in case of mulitple access
139  //gDirectory->ReadKeys();
140  close();
141  if (fMerging ) {
142  // used test merging
143  std::fstream* f = new std::fstream(fname);
144  if (f->good()) {
145  // check if file already exists
146  option = "UPDATE";
147  } else {
148  // if file doesn't exist recreate option is used
149  option = "RECREATE";
150  }
151  f->close();
152  delete f;
153  }
154 
155  file=new FairParRootFile(fname,option,ftitle,compress);
156 
157  if (file && file->IsOpen()) {
158  filename = fname;
159  FairRuntimeDb::instance()->activateParIo(this);
160  return kTRUE;
161  }
162  cerr << "-E- Could not open input file " << fname << endl;
163  Fatal("open","Could not open input file");
164  return kFALSE;
165 }

169 Bool_t FairParRootFileIo::open(const TList* fnamelist, Option_t* option,
170  const Text_t* ftitle, Int_t compress)
171 
172 {
173  TDatime currentDate;
174  TString newParFileName = "";
175  TFile* newParFile=0;
176 
177  TObjString* string;
178  TListIter myIter(fnamelist);
179 
180  TKey* inpKey;
181 
182  TFile* inFile;
183 
184  Int_t nofFiles = 0;
185  while((string = static_cast<TObjString*>(myIter.Next()))) {
186  inFile = TFile::Open(string->GetString().Data());
187  if ( !inFile ) {
188  cout << "-W- File \"" << string->GetString().Data() << "\" does not exist" << endl;
189  continue;
190  }
191 
192  if ( nofFiles == 0 ) {
193  newParFileName = string->GetString();
194  newParFileName.Replace(newParFileName.Last('/')+1,
195  newParFileName.Length(),"");
196  newParFileName = Form("%sallParams_%d_%06d.root",
197  newParFileName.Data(),
198  currentDate.GetDate(),
199  currentDate.GetTime());
200  newParFile = TFile::Open(newParFileName.Data(),"RECREATE");
201  }
202 
203  TList* inputKeys = static_cast<TList*>(inFile->GetListOfKeys());
204 
205  TListIter keyIter(inputKeys);
206  while((inpKey = static_cast<TKey*>(keyIter.Next()))) {
207  TObject* tempObj = inFile->Get(inpKey->GetName());
208 
209  newParFile->cd();
210  tempObj->Write();
211  }
212  inFile->Close();
213 
214  nofFiles++;
215  }
216  if(newParFile!=0){
217  newParFile->Close();
218  }else{
219  std::cout << "****NO file to close file = \"" << std::endl;
220  }
221  std::cout << "**** merged file = \"" << newParFileName.Data() << "\"" << std::endl;
222 
223  return this->open(newParFileName,option,ftitle,compress);
224 }
