129 std::vector<FairTimeStamp*> FairWriteoutBuffer::GetRemoveOldData(double time)
130 {
131  typedef std::multimap<double, FairTimeStamp*>::iterator DTMapIter;
132  std::vector<FairTimeStamp*> result;
133  for(DTMapIter it = fDeadTime_map.begin(); it != fDeadTime_map.lower_bound(time); it++) {
134  if (fVerbose > 1) {
135  std::cout << "-I- GetRemoveOldData: DeadTime: " << it->first << " Data: " << it->second << std::endl;
136  }
137  result.push_back(it->second);
138  EraseDataFromDataMap(it->second);
139  }
140  fDeadTime_map.erase(fDeadTime_map.begin(), fDeadTime_map.lower_bound(time));
141  return result;
142 }

168 void FairWriteoutBuffer::FillDataToDeadTimeMap(FairTimeStamp* data, double activeTime, double startTime)
169 {
170  if (fActivateBuffering) {
171  typedef std::multimap<double, FairTimeStamp*>::iterator DTMapIter;
172 // typedef std::map<FairTimeStamp, double>::iterator DataMapIter;
173  if (activeTime < 0) activeTime = 0; // to avoid errors due to wrong (negative) returnvalues of overwritten modify function by subdetector groups. A negative (-1) aktiveTime already indicate empty data maps!
174  double timeOfOldData = FindTimeForData(data);
175  // PrintDeadTimeMap();
176  if(timeOfOldData > -1) { //if an older active data object is already present
177  if (fVerbose > 1) {
178  std::cout << " OldData found! " << timeOfOldData << std::endl;
179  }
180  if (fVerbose > 1) {
181  std::cout << "New Data: " << activeTime << " : " << data << std::endl;
182  }
183  double currentdeadtime = timeOfOldData;
184  FairTimeStamp* oldData;
185  bool dataFound = false;
186 
187  // PrintDeadTimeMap();
188  for (DTMapIter it = fDeadTime_map.lower_bound(currentdeadtime); it != fDeadTime_map.upper_bound(currentdeadtime); it++) {
189  oldData = it->second;
190  if (fVerbose > 1) {
191  std::cout << "Check Data: " << it->first << " : " << oldData << std::endl;
192  }
193  if (oldData->equal(data)) {
194  if (fVerbose > 1) {
195  std::cout << " oldData == data " << std::endl;
196  }
197  if (fVerbose > 1) {
198  std::cout << it->first << " : " << it->second << std::endl;
199  }
200  fDeadTime_map.erase(it);
201  EraseDataFromDataMap(oldData);
202  dataFound = true;
203  break;
204  }
205  }
206 
207  if (dataFound == true) {
208  if (timeOfOldData > startTime) { //if older active data can interference with the new data call modify function
209  std::vector<std::pair<double, FairTimeStamp*> > modifiedData = Modify(std::pair<double, FairTimeStamp*>(currentdeadtime, oldData), std::pair<double, FairTimeStamp*>(activeTime, data));
210  for (unsigned int i = 0; i < modifiedData.size(); i++) {
211  FillDataToDeadTimeMap(modifiedData[i].second, modifiedData[i].first,0);//startTime = 0 since the Maps are already empty and the startTime therefore doesn't matter anymore
212  if (fVerbose > 1) {
213  std::cout << i << " :Modified Data: " << modifiedData[i].first << " : " << modifiedData[i].second << std::endl;
214  }
215  }
216  } else { //no interference can happen between old hit and new hit
217  AddNewDataToTClonesArray(oldData); //therefore the old hit is written out
218  fDeadTime_map.insert(std::pair<double, FairTimeStamp*>(activeTime, data)); //and the new hit is stored
219  FillDataMap(data, activeTime);
220  }
221  } else {
222  std::cout << "-E- FairWriteoutBuffer::FillDataToDeadTimeMap: old data present in dataMap but not in deadTimeMap!" << std::endl;
223  }
224 
225  } else {
226  if (fVerbose > 1) {
227  std::cout << "-I- FairWriteoutBuffer::FillDataToDeadTimeMap Data Inserted: " << activeTime << " : ";
228  data->Print();
229  std::cout << std::endl;
230  }
231  fDeadTime_map.insert(std::pair<double, FairTimeStamp*>(activeTime, data));
232  FillDataMap(data, activeTime);
233  }
234  } else {
235  AddNewDataToTClonesArray(data);
236  delete data;
237  }
238 }

241 void FairWriteoutBuffer::MoveDataFromStartTimeMapToDeadTimeMap(double time)
242 {
243  typedef std::multimap<double, std::pair<double, FairTimeStamp*> >::iterator startTimeMapIter;
244 
245  startTimeMapIter stopTime = fStartTime_map.lower_bound(time);
246  for (startTimeMapIter iter = fStartTime_map.begin(); iter != stopTime; iter++) {
247  std::pair<double, FairTimeStamp*> data = iter->second;
248  double startTime = iter->first;
249  FillDataToDeadTimeMap(data.second, data.first, startTime);
250  }
251  fStartTime_map.erase(fStartTime_map.begin(), stopTime);
252 }
