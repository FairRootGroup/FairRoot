163  bool CheckVerboseOutput(std::string logLevel, std::string outputString,
164  std::string verboseLevel, std::string resultString
165  ) {
166  // Example Output for logging level INFO and different verbosity levels
167  // verbosity Level:Output
168  // LOW :[INFO ] I am here.
169  // MEDIUM :[INFO ] [_TestFairTools.cxx::LogNoArguments:78] I am here.
170  // HIGH :[INFO ] [05.12.2011 12:00:10] [_TestFairTools.cxx::LogNoArguments:78] I am here.
171  std::string regexString = "\\[" + logLevel + "[ ]*][ ]";
172  if (verboseLevel.compare("HIGH") == 0) {
173  regexString = regexString + "\\[\\d{2}\\.\\d{2}\\.\\d{4}[ ]\\d{2}:\\d{2}:\\d{2}][ ]";
174  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
175  }
176  if (verboseLevel.compare("MEDIUM") == 0) {
177  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
178  }
179  regexString = regexString + outputString + ".*";
180 
181  const std::regex e(regexString);
182  return std::regex_match(resultString, e);
183  }
