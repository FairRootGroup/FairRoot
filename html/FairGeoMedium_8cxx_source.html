87 void FairGeoMedium::setNComponents(Int_t n)
88 {
89  // Sets the number of components in the material
90  if (n==0) { return; }
91  Int_t k=abs(n);
92  if (nComponents!=0 && k!=nComponents) {
93  delete [] ca;
94  delete [] cz;
95  delete [] cw;
96  nComponents=0;
97  }
98  if (nComponents==0) {
99  nComponents=k;
100  ca=new Double_t[k];
101  cz=new Double_t[k];
102  cw=new Double_t[k];
103  }
104  weightFac=static_cast<Int_t>(n/nComponents);
105 }

204 void FairGeoMedium::read(std::fstream& fin, Int_t aflag )
205 {
206  // Reads the parameters from file
207  autoflag=aflag;
208  Int_t n;
209  fin>>n;
210  setNComponents(n);
211  for(Int_t ik=0; ik<nComponents; ik++) {
212  fin>>ca[ik];
213  }
214  for(Int_t i=0; i<nComponents; i++) {
215  fin>>cz[i];
216  }
217  fin>>density;
218  if (nComponents==1) {
219  cw[0]=1.;
220  calcRadiationLength();
221  } else {
222  for(Int_t i=0; i<nComponents; i++) {
223  fin>>cw[i];
224  }
225  }
226  fin>>sensFlag>>fldFlag>>fld>>epsil ;
227  if (autoflag<1) { fin>>madfld>>maxstep>>maxde>>minstep; }
228  else {
229  //to use this feature one has to set TGeant3::SetAUTO(0), thus if the media does not
230  // defined these values one can force Geant3 to calculate them by given them a value
231  // of -1
232  madfld=-1;
233  maxstep=-1;
234  maxde=-1;
235  minstep=-1;
236  }
237  fin>>n;
238  if (n>0 && n<(INT_MAX-1)) {
239  setNpckov(n);
240 // if (n>0) {
241  for(Int_t i=0; i<n; i++) {
242  fin>>ppckov[i]>>absco[i]>>effic[i]>>rindex[i];
243  }
244  }

294 Bool_t FairGeoMedium::calcRadiationLength()
295 {
296  // calculates radiation length
297  // formula in GEANT manual CONS110
298  if (cz[0]<1.e-6) { // VAKUUM$
299  radLen=1.e+16;
300  return kTRUE;
301  }
302  Double_t alpha=1/137.; // fine structure constant
303  Double_t fac=.1912821; // 4*((electron radius)**2)*(avogadro's number)
304  Double_t z, a, w, az2, fc, y, xi, x0i, amol=0., x0itot=0.;
305  if (weightFac>0) { amol=1.; }
306  else {
307  for (int i=0; i<nComponents; i++) {
308  amol+=cw[i]*ca[i];
309  if (amol==0.) {
310  Error("calcRadiationLength()","amol==0 for medium %s",fName.Data());
311  return kFALSE;
312  }
313  }
314  }
315  for (int i=0; i<nComponents; i++) {
316  z=cz[i];
317  a=ca[i];
318  if(weightFac>0) {
319  w=cw[i]/amol;
320  } else {
321  w=a*cw[i]/amol;
322  }
323  az2=alpha*alpha*z*z;
324  fc=az2 * (1./(1.+az2) + 0.20206 - 0.0369*az2 + 0.0083*az2*az2
325  - .002F*az2*az2*az2);
326  y=log(183./pow(z,1./3.)) - fc;
327  xi=static_cast<float>(log(1440./pow(z,2./3.)) / y);
328  x0i=fac*alpha/a*z*(z+xi)*y;
329  x0itot+=(x0i*w);
330  }
331  if (x0itot==0. || density==0.) {
332  Error("calcRadiationLength()","x0itot=0 or density=0 for medium %s",fName.Data());
333  return kFALSE;
334  }
335  radLen=1/density/x0itot;
336  return kTRUE;
337 }
