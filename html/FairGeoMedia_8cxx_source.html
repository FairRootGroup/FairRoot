101 void FairGeoMedia::read(std::fstream& fin)
102 {
103  // Reads the media from file
104  LOG(info) << "FairGeoMedia: Read media ";
105  const Int_t maxBuf=256;
106  char buf[maxBuf];
107  Int_t autoflag=1;
108  while(!fin.eof()) {
109  fin>>buf;
110  if (buf[0]=='\0' || buf[0]=='/') { fin.getline(buf,maxBuf); }
111  else if (fin.eof()) { break; }
112  else {
113  TString eleName(buf);
114  if (eleName.CompareTo("AUTONULL")!=0) {
115  FairGeoMedium* medium=new FairGeoMedium(eleName);
116  medium->read(fin,autoflag);
117  media->Add(medium);
118  } else { autoflag=0; }
119  }
120  }
121 }
