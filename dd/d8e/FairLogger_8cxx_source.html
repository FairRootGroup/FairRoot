166 void FairLogger::Log(FairLogLevel level, const char* file, const char* line,
167  const char* func, const char* format, va_list arglist)
168 {
169  // If the format string together with the argument list was used once it can't
170  // be used another time. Don't know why but if we do so the arguments are not
171  // correct any longer and the program may crash (On Mac OSX). Even if it does
172  // not crash it does not work as expected.
173  // To vercome the problem the output is written to a buffer which then can be
174  // used several times.
175 
176  if(fLogToFile && !fLogFileOpen) {
177  OpenLogFile();
178  }
179 
180  fLevel = level;
181  while (1) {
182  fBufferSizeNeeded = vsnprintf(fBufferPointer, fBufferSize, format, arglist);
183 
184  // Try to vsnprintf into our buffer.
185  // int needed = vsnprintf (buf, size, fmt, ap);
186  // NB. C99 (which modern Linux and OS X follow) says vsnprintf
187  // failure returns the length it would have needed. But older
188  // glibc and current Windows return -1 for failure, i.e., not
189  // telling us how much was needed.
190 
191  if (fBufferSizeNeeded <= (int)fBufferSize && fBufferSizeNeeded >= 0) {
192  // It fit fine so we're done.
193  GetOutputStream(level, file, line, func) <<
194  std::string(fBufferPointer, (size_t) fBufferSizeNeeded)<<
195  " " << FairLogger::endl;
196 
197  break;
198  }
199 
200  // vsnprintf reported that it wanted to write more characters
201  // than we allotted. So try again using a dynamic buffer. This
202  // doesn't happen very often if we chose our initial size well.
203  fBufferSize = (fBufferSizeNeeded > 0)
204  ? (fBufferSizeNeeded+1) : (fBufferSize*2);
205 
206  fDynamicBuffer.resize (fBufferSize);
207  }
208 }

229 void FairLogger::OpenLogFile()
230 {
231  // Check if the log file name is set. If not use a default logfile with
232  // the name FairLogfile_<pid>.log. <pid> is the pid of the runing job.
233  TString logfile = fLogFileName;
234  if (logfile.IsNull()) {
235  Int_t PID = gSystem->GetPid();
236  logfile = "FairLogfile_";
237  logfile += PID;
238  logfile += ".log";
239 
240  }
241 
242  // Check if filename contains a full path. If not create logfile
243  // in the current directory.
244  // TODO(F.U.): Check if the path exists and is writeable.
245  if (!logfile.Contains('/')) {
246  logfile = "./" + logfile;
247  }
248 
249  if (fFileStream) {
250  CloseLogFile();
251  delete fFileStream;
252  fFileStream = NULL;
253  }
254 
255  fFileStream = new std::ofstream(logfile.Data());
256 
257  fLogFileOpen = kTRUE;
258 }

260 FairLogLevel FairLogger::ConvertToLogLevel(const char* levelc) const
261 {
262  // Convert string to upper case
263  // Return the according log level
264  // in case the level is not known return info level
265  TString level = levelc;
266  level.ToUpper();
267  if (level == "FATAL") { return FATAL; }
268  if (level == "ERROR") { return ERROR; }
269  if (level == "WARNING") { return WARNING; }
270  if (level == "INFO") { return INFO; }
271  if (level == "DEBUG") { return DEBUG; }
272  if (level == "DEBUG1") { return DEBUG1; }
273  if (level == "DEBUG2") { return DEBUG2; }
274  if (level == "DEBUG3") { return DEBUG3; }
275  if (level == "DEBUG4") { return DEBUG4; }
276  LOG(ERROR)<<"Log level \""<<level<<"\" not supported. Use default level \"INFO\"."<<FairLogger::endl;
277  return INFO;
278 }

319 {
320 
321  fLevel = level;
322 
323  if (level == FATAL) {
324  fLogToScreen = true;
325  fLogVerbosityLevel = verbosityHIGH;
326  fLogColored = true;
327  }
328 
329 
330  if (fIsNewLine) {
331  if ( (fLogToScreen && level <= fLogScreenLevel) ) {
332 
333  if ( fLogColored ) {

338 
339  if ( fLogVerbosityLevel == verbosityHIGH ) {

343

354  if(!fLogFileOpen) {
355  OpenLogFile();
356  }
357 
358  *fFileStream << "[" << std::setw(7) << std::left << LogLevelString[level] <<"] ";
359 
360  if ( fLogVerbosityLevel == verbosityHIGH ) {
361  GetTime();
362  *fFileStream << fTimeBuffer;
363  }
364 
365  if ( fLogVerbosityLevel <= verbosityMEDIUM ) {
366  TString bla(file);

368  TString s2(bla(pos+1, bla.Length()));

443 void FairLogger::LogFatalMessage(std::ostream& strm)
444 {
445  // Since Fatal indicates a fatal error it is maybe usefull to have
446  // system information from the incident. Since the output on the screen
447  // does not help the noraml user, the stderr is redirected into a special
448  // core dump file for later usage.
449  // Fatal also indicates a problem which is so severe that the process should
450  // not go on, so the process is aborted.
451 
452  *fScreenStream << LogLevelColor[FATAL];
453 
454  *fScreenStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
455 
456  GetTime();
457  *fScreenStream << fTimeBuffer;
458 
459  *fScreenStream << "We stop the execution of the process at this point.\n";
460 
461  if ( fLogToFile ) {
462  *fFileStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
463 
464  *fFileStream << fTimeBuffer;
465  *fFileStream << "We stop the execution of the process at this point.\n";
466  }
467 
468  if (gSystem) {
469  TString corefile = "core_dump_";
470  Int_t PID = gSystem->GetPid();
471  corefile += PID;
472 
473 
474  *fScreenStream << LogLevelColor[FATAL];
475 
476  *fScreenStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
477 
478  GetTime();
479  *fScreenStream << fTimeBuffer;
480 
481  *fScreenStream << "For later analysis we write a core dump to " <<
482  corefile << " \n";
483 
484 
485  if ( fLogToFile ) {
486  *fFileStream << "[" << std::setw(7) << std::left << LogLevelString[FATAL] <<"] ";
487 
488  *fFileStream << fTimeBuffer;
489  *fFileStream << "For later analysis we write a core dump to" <<
490  corefile << "\n";
491  }
492 
493  *(gLogger->fScreenStream) << "\33[00;30m" << std::endl;
494 
495  flush(strm);
496  freopen(corefile, "w", stderr);
497  gSystem->StackTrace();
498  fclose(stderr);
499  gSystem->Abort(1);
500  } else {
501  abort();
502  }
503 
504 }
