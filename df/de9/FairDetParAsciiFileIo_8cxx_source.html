100 void FairDetParAsciiFileIo::readComment(const Char_t* lastBuf, FairParSet* pPar)
101 {
102  // reads the author or the description
103  const Int_t maxbuf=4000;
104  Text_t buf[maxbuf];
105  TString s=lastBuf;
106  TString pName, pVal;
107  Ssiz_t n=s.First(':');
108  Ssiz_t m=s.Last('\\');
109  pName=s(0,n);
110  if (m>0) { pVal=s(n+1,m-n-1); }
111  else { pVal=s(n+1,s.Length()-n-1); }
112  pVal=pVal.Strip(pVal.kLeading);
113  while (m>0) {
114  pVal+="\n";
115  pFile->getline(buf,maxbuf);
116  if (buf[0]!='/') {
117  s=buf;
118  m=s.Last('\\');
119  if (m>0) { pVal+=s(0,m); }
120  else { pVal+=s; }
121  }
122  }
123  if (pName.CompareTo("author")==0) { pPar->setAuthor(pVal.Data()); }
124  else if ( pName.CompareTo("description")==0) { pPar->setDescription(pVal.Data()); }
125 }
