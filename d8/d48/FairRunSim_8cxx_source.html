143 void FairRunSim::Init()
144 {
147  CheckFlukaExec();
148 
149 // fOutFile=fRootManager->OpenOutFile(fOutname);
150  LOG(INFO) << "============== FairRunSim: Initialising simulation run ==============" << FairLogger::endl;
151 
152  FairGeoLoader* loader=new FairGeoLoader(fLoaderName->Data(), "Geo Loader");
153  FairGeoInterface* GeoInterFace=loader->getGeoInterface();
154  GeoInterFace->SetNoOfSets(ListOfModules->GetEntries());
155  GeoInterFace->setMediaFile(MatFname.Data());
156  GeoInterFace->readMedia();
157 
158  // gSystem->cd(flout.Data());
159 
160  fApp= new FairMCApplication("Fair","The Fair VMC App",ListOfModules, MatFname);
161  fApp->SetGenerator(fGen);
162 
163  // Add a Generated run ID to the FairRunTimeDb
164  FairRunIdGenerator genid;
165  // FairRuntimeDb *rtdb= GetRuntimeDb();
166  fRunId = genid.generateId();
167  fRtdb->addRun(fRunId);
168 
169  fFileHeader->SetRunId(fRunId);
171  fApp->AddTask(fTask);
172 
173  FairBaseParSet* par=dynamic_cast<FairBaseParSet*>(fRtdb->getContainer("FairBaseParSet"));
174  if (par) {
175  par->SetDetList(GetListOfModules());
176  par->SetGen(GetPrimaryGenerator());
177  par->SetBeamMom(fBeamMom);
178  }
179 
180  FairGeoParSet* geopar=dynamic_cast<FairGeoParSet*>(fRtdb->getContainer("FairGeoParSet"));
181  if (geopar) {
182  geopar->SetGeometry(gGeoManager);
183  }
184 
185  // Set global Parameter Info
186 
187  if(fPythiaDecayer) {
188  fApp->SetPythiaDecayer(fPythiaDecayer);
189  if (fPythiaDecayerConfig) {
190  fApp->SetPythiaDecayerConfig(fPythiaDecayerConfig);
191  }
192  }
193  if(fUserDecay) {
194  fApp->SetUserDecay(fUserDecay);
195  if (fUserDecayConfig) {
196  fApp->SetUserDecayConfig(fUserDecayConfig);
197  }
198  }
199  // on/off visualisation
200  if( fStoreTraj ) {
201  LOG(INFO) << "Create visualisation manager " << FairLogger::endl;
202  new FairTrajFilter();
203  }
204  if(fRadLength) {
205  fApp->SetRadiationLengthReg(fRadLength);
206  }
207  if(fRadMap) {
208  fApp->SetRadiationMapReg(fRadMap);
209  }
210  if(fRadGrid) {
211  fApp->AddMeshList(fMeshList);
212  }
213 
214  if(fField) { fField->Init(); }
215  fApp->SetField(fField);
216  SetFieldContainer();
217 
218  TList* containerList=fRtdb->getListOfContainers();
219  TIter next(containerList);
220  FairParSet* cont;
221  TObjArray* ContList= new TObjArray();
222  while ((cont=dynamic_cast<FairParSet*>(next()))) {
223  ContList->Add(new TObjString(cont->GetName()));
224  }
225 
226  par->SetContListStr(ContList);
227  par->setChanged();
228  par->setInputVersion(fRunId,1);
229 
230  geopar->setChanged();
231  geopar->setInputVersion(fRunId,1);
232 
233 
234 
236  SetMCConfig();
237  fRootManager->WriteFileHeader(fFileHeader);
238 
239 }

248 void FairRunSim::CheckFlukaExec()
249 {
251  TString work = getenv("VMCWORKDIR");
252  TString work_config=work+"/gconfig/";
253  work_config.ReplaceAll("//","/");
254 
255  TString config_dir= getenv("CONFIG_DIR");
256  config_dir.ReplaceAll("//","/");
257  if (!config_dir.EndsWith("/")) { config_dir+="/"; }
258 
259  TString flout;
260  if(strcmp(GetName(),"TFluka") == 0 ) {
261  TString flexec="run_fluka.sh";
262  if (TString(gSystem->FindFile(config_dir.Data(),flexec)) != TString("")) {
263  LOG(INFO) << "---User path for Configuration is used: "
264  << config_dir.Data() << FairLogger::endl;
265  } else {
266  flexec=work_config+"run_fluka.sh";
267  }
268  gSystem->Exec(flexec.Data());
269  // More generic for different macro
270  // dir layout <D.B>
271  TString macrodir = getenv("PWD");
272  cout << macrodir << endl;
273  flout = macrodir + "/fluka_out" ;
274  flout.ReplaceAll("//","/");
275  gSystem->cd(flout.Data());
276  }

280 {

286

293 
294  Bool_t AbsPath=kFALSE;
295 
296  TString LibMacro;
297  TString LibFunction;
298  TString ConfigMacro;
299  TString cuts=fUserCuts;
300  //----------------------------------------------Geant4 Config-----------------------------------------
301  if(strcmp(GetName(),"TGeant4") == 0 ) {
302  TString g4LibMacro="g4libs.C";
303  TString g4Macro;
304  if(fUserConfig.IsNull()) {
305  g4Macro="g4Config.C";
306  fUserConfig = g4Macro;
307  } else {
308  if (fUserConfig.Contains("/")) { AbsPath=kTRUE; }
309  g4Macro = fUserConfig;

311  << g4Macro.Data() << FairLogger::endl;
312  }
313  if (TString(gSystem->FindFile(config_dir.Data(),g4LibMacro)) != TString("")) { //be carfull after this call the string g4LibMacro is empty if not found!!!!
314  LOG(INFO) << "---User path for Configuration (g4libs.C) is used: "
315  << config_dir.Data() << FairLogger::endl;
316  } else {
317  g4LibMacro=Lib_config+"g4libs.C";
318  }
319  LibMacro=g4LibMacro;
320  LibFunction="g4libs()";
321  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),g4Macro)) != TString("")) {
322  LOG(INFO) << "---User path for Configuration (g4Config.C) is used: "
323  << config_dir.Data() << FairLogger::endl;
324  ConfigMacro=g4Macro;
325  } else {
326  if(AbsPath) { ConfigMacro = fUserConfig; }

347  g3LibMacro=work_config+"g3libs.C";

352  LOG(INFO) << "---User path for Configuration (g3Config.C) is used: "
353  << config_dir.Data() << FairLogger::endl;
354  ConfigMacro=g3Macro;
355  } else {
356  if(AbsPath) { ConfigMacro = fUserConfig; }
357  else { ConfigMacro =work_config+fUserConfig; }
358  }
359  //----------------------------------------------Fluka Config-----------------------------------------
360  } else if(strcmp(GetName(),"TFluka") == 0 ) {
361  TString flLibMacro="fllibs.C";
362  TString flMacro="flConfig.C";
363  if(fUserConfig.IsNull()) {
364  flMacro="flConfig.C";
365  fUserConfig=flMacro;
366  } else {
367  if (fUserConfig.Contains("/")) { AbsPath=kTRUE; }

373  LOG(INFO) << "---User path for Configuration (fllibs.C) is used: "

376  flLibMacro=work_config+"fllibs.C";
377  }
378  LibMacro=flLibMacro;
379  LibFunction="fllibs()";
380  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),flMacro)) != TString("")) {
381  LOG(INFO) << "---User path for Configuration (flConfig.C) is used: "
382  << config_dir.Data() << FairLogger::endl;
383  ConfigMacro=flMacro;
384  } else {
385  if(AbsPath) { ConfigMacro = fUserConfig; }
386  else { ConfigMacro =work_config+fUserConfig; }
387  }
388  }
389  //----------------------------------------------SetCuts------------------------------------------------
390  if (TString(gSystem->FindFile(config_dir.Data(),cuts)) != TString("")) {
391  LOG(INFO) << "---User path for Cuts and Processes (SetCuts.C) is used: "
392  << config_dir.Data() << FairLogger::endl;
393  } else {
394  cuts =work_config+ fUserCuts;
395  }
396  //--------------------------------------Now load the Config and Cuts------------------------------------
397  gROOT->LoadMacro(LibMacro.Data());
398  gROOT->ProcessLine(LibFunction.Data());
399 
400  gROOT->LoadMacro(ConfigMacro.Data());
401  gROOT->ProcessLine("Config()");
402 
403  gROOT->LoadMacro(cuts);
404  gROOT->ProcessLine("SetCuts()");
405 
406  fApp->InitMC(ConfigMacro.Data(), cuts.Data());
407 }
