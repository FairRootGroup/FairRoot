415 void FairRootManager::Register(const char* name, const char* folderName , TNamed* obj, Bool_t toFile)
416 {
417 
418  // Security check. If the the name is equal the folder name there are problems with reading
419  // back the data. Instead of the object inside the folder the RootManger will return a pointer
420  // to the folder. To avoid such problems we check here if both strings are equal and stop the
421  // execution with some error message if this is the case.
422  if (strcmp (name, folderName) == 0 ) {
423  fLogger->Fatal(MESSAGE_ORIGIN,"The names for the object name %s and the folder name %s are equal. This isn't allowed. So we stop the execution at this point. Pleae change either the name or the folder name.", name, folderName);
424  }
425 
426  if(toFile) { /**Write the Object to the Tree*/
427  TFolder* folder=0;
428  TFolder* f=0;
429  if(fCbmout==0) {
430  f=(TFolder*)fCbmroot->FindObjectAny(folderName);
431  if(f==0) {
432  folder= fCbmroot->AddFolder(folderName,folderName);
433  } else {
434  folder=f;
435  }
436  } else {
437  f=(TFolder*)fCbmout->FindObjectAny(folderName);
438  if(f==0) {
439  folder= fCbmout->AddFolder(folderName,folderName);
440  } else {
441  folder=f;
442  }
443  }
444  obj->SetName(name);
445  folder->Add(obj);
446  }
447  AddMemoryBranch(name, obj );
448  //cout << " FairRootManager::Register Adding branch:(Obj) " << name << " In folder : " << folderName << endl;
449  if(fBranchNameList->FindObject(name)==0) {
450  fBranchNameList->AddLast(new TObjString(name));
451  fBranchSeqId++;
452  }
453 
454  if (toFile == kFALSE) {
455  FairLinkManager::Instance()->AddIgnoreType(GetBranchId(name));
456  }
457 }

532 TClonesArray* FairRootManager::GetEmptyTClonesArray(TString branchName)
533 {
534  if (fActiveContainer.find(branchName) != fActiveContainer.end()) { //if a TClonesArray is registered in the active container
535  if (fActiveContainer[branchName] == 0) { //the address of the TClonesArray is still valid
536  std::cout << "-E- FairRootManager::GetEmptyTClonesArray: Container deleted outside FairRootManager!" << std::endl;
537  } else if (!fCompressData) { //if the data is not compressed the existing TClonesArray is just emptied
538  fActiveContainer[branchName]->Delete();
539  } else if (fActiveContainer[branchName]->GetEntries() > 0) { //if the container is not empty push it into the DataContainer storage and create a new one
540  fDataContainer[branchName].push(fActiveContainer[branchName]);
541  fLogger->Info(MESSAGE_ORIGIN, "GetEmptyTClonesArray moved %s with %i to data container ",
542  branchName.Data(), fActiveContainer[branchName]->GetEntries());
543  fActiveContainer[branchName] = new TClonesArray(fActiveContainer[branchName]->GetClass()->GetName());
544  } else {
545  fLogger->Info(MESSAGE_ORIGIN, "GetEmptyTClonesArray not moved %s %s with %i to data container ",
546  branchName.Data(),fActiveContainer[branchName] , fActiveContainer[branchName]->GetEntries());
547  }
548  return fActiveContainer[branchName]; // return the container
549  } else {
550  std::cout << "-E- Branch: " << branchName << " not registered!" << std::endl; // error if the branch is not registered
551  }
552  return 0;
553 }
