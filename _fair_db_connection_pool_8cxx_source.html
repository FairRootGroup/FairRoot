218 Int_t FairDbConnectionPool::CreateTemporaryTable(const string& tableNameMc,
219  const string& tableDescr)
220 {
221 
222  string tableName = FairUtilString::ToUpper(tableNameMc);
223  if ( tableName == ""
224  || tableDescr[0] != '('
225  || tableDescr[tableDescr.size()-1] != ')' ) {
226  MAXDBLOG("FairDb",FairDbLog::kError,20) << "Illegal input args:-" << endl
227  << " Table Name: " << tableName
228  << " Table Description: " << tableDescr
229  <<endl;
230  return -1;
231  }
232 
233 // Find a DB that will accept the command.
234  string sqlMakeTable;
235 
236  Int_t dbNoAcc = -1;
237  auto_ptr<FairDbStatement> stmtDb;
238  for (UInt_t dbNoTry = 0; dbNoTry < fConnections.size(); ++dbNoTry ) {
239  stmtDb.reset(this->CreateStatement(dbNoTry));
240  if ( stmtDb.get() ) {
241  sqlMakeTable = " create temporary table ";
242  sqlMakeTable += tableName;
243  sqlMakeTable += " ";
244  sqlMakeTable += tableDescr;
245  sqlMakeTable += ";";
246  stmtDb->ExecuteUpdate(sqlMakeTable.c_str());
247  if ( stmtDb->GetExceptionLog().IsEmpty() ) {
248  dbNoAcc = dbNoTry;
249  this->GetConnection(dbNoAcc)->SetTableExists(tableName);
250  break;
251  }
252  // Print exceptions if needed
253  if (FairDbLogService::Instance()->IsActive("FairDb",FairDbLog::kWarning)) { stmtDb->PrintExceptions(); }
254  }
255  }
256 
257  if ( dbNoAcc < 0 ) {
258  if ( stmtDb.get()) { stmtDb->PrintExceptions(); }
259  return -1;
260  }
261 
262 // Make connection permanent if not already.
263  FairDbConnection& conDb = *fConnections[dbNoAcc];
264  if ( conDb.IsTemporary() ) {
265  conDb.SetPermanent();
266  DBLOG("FairDb",FairDbLog::kWarning) << "Making connection: " << conDb.GetUrl()
267  << " permanent to preserve temporary tables." << endl;
268  }
269 
270 // Create SQL to create auxillary validity table and write to same Db.
271  sqlMakeTable = FairDb::GetValDefinition(tableName.c_str(),true);
272 
273  DBLOG("FairDb",FairDbLog::kWarning) << "Validity Table creation: "
274  << " Database: " << dbNoAcc << " "
275  << sqlMakeTable << endl;
276  stmtDb->ExecuteUpdate(sqlMakeTable.c_str());
277  if ( stmtDb->PrintExceptions() ) { return -1; }
278  this->GetConnection(dbNoAcc)->SetTableExists(tableName+"VAL");
279  fTemporaryTables[tableName] = dbNoAcc;
280  return dbNoAcc;
281

439 Int_t FairDbConnectionPool::ReserveNextSeqNo(const string& tableName,
440  Bool_t isGlobal,
441  UInt_t dbNo) const
442 {
443 
444  FairDbString sql;
445 
446  string seqnoTableName = isGlobal ? "FAIRDBGLOBALSEQNO" : "FAIRDBLOCALSEQNO";
447  bool seqnoTableNameExists = this->TableExists(seqnoTableName,dbNo);
448  bool tableNameExists = this->TableExists(tableName,dbNo);
449 
450  auto_ptr<FairDbStatement> stmtDb(this->CreateStatement(dbNo) );
451  if ( ! stmtDb.get() ) { return 0; }
452 
453  // Check that required SEQNO table exists.
454  if ( isGlobal ) {
455  if ( ! seqnoTableNameExists ) {
456  DBLOG("FairDb",FairDbLog::kInfo)<< "Unable to issue global SEQNO - " << dbNo
457  << " is not an authorising DB" << endl;
458  return 0;
459  }
460  } else {
461  if ( ! seqnoTableNameExists ) {
462  sql.Clear();
463  sql << "CREATE TABLE " << seqnoTableName
464  << "(TABLENAME CHAR(64) NOT NULL PRIMARY KEY,\n"
465  << " LASTUSEDSEQNO INT )";
466 
467  DBLOG("FairDb",FairDbLog::kInfo)<< "Database: " << dbNo
468  << " create local SEQNO table query: " << sql.c_str() << endl;
469  stmtDb->ExecuteUpdate(sql.c_str());
470  if ( stmtDb->PrintExceptions() ) { return 0; }
471  sql.Clear();
472  sql << "INSERT INTO " << seqnoTableName << " VALUES ('*',0)";
473  DBLOG("FairDb",FairDbLog::kInfo)<< "Database: " << dbNo
474  << " prime local SEQNO table query: " << sql.c_str() << endl;
475  stmtDb->ExecuteUpdate(sql.c_str());
476  if ( stmtDb->PrintExceptions() ) { return 0; }
477  }
478  }
479 
480 
481  string dataTable;
482 
483  if ( ! this->IsTemporaryTable(tableName,dbNo)
484  && tableNameExists ) { dataTable = tableName; }
485  BLock Block(this->CreateStatement(dbNo),seqnoTableName,dataTable);
486  if ( ! Block.IsBLocked() ) {
487  DBLOG("FairDb",FairDbLog::kInfo)<< "Unable to lock " << seqnoTableName << endl;
488  return 0;
489  }
490 
491 
492  sql.Clear();
493  sql << "select * from " << seqnoTableName << " where TABLENAME = '*' or TABLENAME = '";
494  sql << tableName + "' order by TABLENAME";
495  DBLOG("FairDb",FairDbLog::kInfo)<< " tablename: "<< seqnoTableName << " query: " << sql.c_str() << endl;
496  TSQLStatement* stmt = stmtDb->ExecuteQuery(sql.c_str());
497  stmtDb->PrintExceptions(FairDbLog::kDebug);
498  Int_t seqNoDefault = 0;
499  if ( stmt && stmt->NextResultRow() ) {
500  seqNoDefault = stmt->GetInt(1);
501  } else {
502  DBLOG("FairDb",FairDbLog::kInfo)<< "Unable to find default SeqNo"
503  << " due to above error" << endl;
504  delete stmt;
505  stmt = 0;
506  return 0;
507  }
508  Int_t seqNoTable = seqNoDefault;
509  if ( stmt->NextResultRow() ) {
510  seqNoTable = stmt->GetInt(1);
511  }
512  delete stmt;
513  stmt = 0;
514  DBLOG("FairDb",FairDbLog::kInfo)<< "query returned last generated seqno: " << seqNoTable << endl;
515 
516 
517  static std::string checkedCombinations;
518  ostringstream combination;
519  combination << ":" << tableName << isGlobal << dbNo << ":";
520  bool notChecked = checkedCombinations.find(combination.str()) == std::string::npos;
521  if ( notChecked ) { checkedCombinations += combination.str(); }
522  if ( tableNameExists && notChecked ) {
523  Int_t seqNoMin = seqNoDefault;
524  Int_t seqNoMax = seqNoDefault + FairDb::kMAXLOCALSEQNO;
525  sql.Clear();
526  sql << "select max(SEQNO) from " << tableName << "VAL"
527  << " where SEQNO between " << seqNoMin << " and " << seqNoMax;
528  DBLOG("FairDb",FairDbLog::kInfo)<< "Database: " << dbNo
529  << " max SEQNO query: " << sql.c_str() << endl;
530  stmt = stmtDb->ExecuteQuery(sql.c_str());
531  if ( stmtDb->PrintExceptions() ) { return 0; }
532  Int_t minValue = 0;
533  // Queries returning group function results can be null.
534  if ( stmt && stmt->NextResultRow() && ! stmt->IsNull(0) ) {
535  minValue = stmt->GetInt(0);
536  if ( minValue <= 0 ) { minValue = 0; } // Should never happen.
537  }
538  delete stmt;
539  stmt = 0;
540 
541  if ( minValue > seqNoTable ) {
542  DBLOG("FairDb",FairDbLog::kInfo)
543  << "Database: " << dbNo << " "
544  << seqnoTableName << " has last used SEQNO of "
545  << seqNoTable << " for table " << tableName
546  << ",\n but the highest SEQNO in the band " << seqNoMin << " to " << seqNoMax
547  << " is " << minValue << " for that table\n "
548  << seqnoTableName << " is out of date! It will be updated for " << tableName << endl;
549  seqNoTable = minValue;
550  }
551  }
552 
553 
554  sql.Clear();
555  sql << "delete from " << seqnoTableName << " where TABLENAME='";
556  sql << tableName + "'";
557  DBLOG("FairDb",FairDbLog::kInfo)<< "SEQNO entry removal: "
558  << sql.c_str() << endl;
559  stmtDb->ExecuteUpdate(sql.c_str());
560  if ( stmtDb->PrintExceptions() ) { return 0; }
561 
562  seqNoTable++;
563 
564  sql.Clear();
565  sql << "insert into " << seqnoTableName << " values('";
566  sql << tableName + "'," << seqNoTable << ")";
567  DBLOG("FairDb",FairDbLog::kInfo)<< "SEQNO entry add: "
568  << sql.c_str() << endl;
569  stmtDb->ExecuteUpdate(sql.c_str());
570  if ( stmtDb->PrintExceptions() ) { return 0; }
571 
572  return seqNoTable;
573
