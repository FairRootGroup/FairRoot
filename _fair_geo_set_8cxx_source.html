72 void FairGeoSet::setModules(Int_t s,Int_t* m)
73 {
74  // Stores the modules given in 'm' as active modules in sector 's'
75  // May be called with s=-1 for detectors not belonging to a sector
76  if (!modules) {
77  if (maxSectors==0&&maxModules>0) { modules=new TArrayI(maxModules); }
78  if (maxSectors>0&&maxModules>0) { modules=new TArrayI(maxSectors*maxModules); }
79  }
80  Int_t sec=0;
81  if (s>0) { sec=s; }
82  for(Int_t i=0; i<maxModules; i++) {
83  if (modules) { modules->AddAt(m[i],(sec*maxModules+i)); }
84  }
85 }

94 Int_t FairGeoSet::getModule(Int_t s,Int_t m)
95 {
96  // Returns 1, if the module is not explicitly set inactive
97  // (in this case it returns 0)
98  // May be called with s=-1 for detectors not belonging to a sector
99  if (!modules) { return 1; }
100  if (m<maxModules&&s<0) { return (*modules)[m]; }
101  if (m<maxModules&&s<maxSectors) { return (*modules)[(s*maxModules+m)]; }
102  return 0;
103 }

105 Bool_t FairGeoSet::read(fstream& fin,FairGeoMedia* media)
106 {
107  // Reads the geometry from file
108  Int_t s1=-1,s2=0;
109  if (maxSectors>0) {
110  s1=0;
111  s2=maxSectors;
112  }
113  Bool_t rc=kTRUE;
114  for (Int_t s=s1; s<s2&&rc; s++) {
115  for (Int_t k=0; k<maxKeepinVolumes&&rc; k++) {
116  TString keepinName=getKeepinName(s,k);
117  rc=readKeepIn(fin,media,keepinName);
118  }
119  }
120  for (Int_t m=0; m<maxModules&&rc; m++) {
121  Bool_t containsActiveModule=kFALSE;
122  for (Int_t s=s1; s<s2; s++) {
123  if (getModule(s,m)) { containsActiveModule=kTRUE; }
124  }
125  TString modName=getModuleName(m);
126  TString eleName=getEleName(m);
127  rc=readModule(fin,media,modName,eleName,containsActiveModule);
128  }
129  return rc;
130 }

156 Bool_t FairGeoSet::readVolumeParams(fstream& fin,FairGeoMedia* media,
157  FairGeoNode* volu,TList* refVolumes)
158 {
159  // Reads the volume definition from file
160 // Int_t hadgeo = 1;
161  if (volu==0||media==0||refVolumes==0) { return kFALSE; }
162  const Int_t maxbuf=256;
163  char buf[maxbuf];
164  TString name(volu->GetName());
165  Int_t nameLength=name.Length();
166  Bool_t isCopy=kFALSE;
167  fin>>buf;
168  FairGeoNode* mother=0;
169  FairGeoCopyNode* refNode=0;
170  TString refName;
171 
172  if (volu->isKeepin()) { mother=getMasterNode(buf); }
173  else if (volu->isModule()) {
174  mother=getMasterNode(buf);
175  if (!mother) { mother=getVolume(buf); }
176  if (volu->isActive()&&mother) { mother->setActive(); }
177  } else {
178  mother=getVolume(buf);
179  if (!mother) { mother=getMasterNode(buf); }
180  }
181 
182  volu->setMother(mother);
183  if (!mother) { Warning("readVolumeParams","Mother volume %s not found!",buf); }
184 
185  if ( hadesGeo == 1 ) {
186  // cout << " read copies in Hades format " << endl;
187 
188  if (nameLength>4) {
189  char c;
190  do {
191  c=fin.get();
192  } while (c==' ' || c=='\n');
193  Int_t i=(Int_t)c ;
194  fin.putback(c);
195 
196  if (!isalpha(i)) { isCopy=kTRUE; }
197  refName=name(0,4);
198  refNode=(FairGeoCopyNode*)refVolumes->FindObject(refName);
199  if (!refNode) {
200  if (isCopy) { return kFALSE; }
201  refVolumes->Add(new FairGeoCopyNode(refName.Data(),volu));
202  } else {
203  FairGeoNode* cn=refNode->pNode;
204  volu->setCopyNode(cn);
205  volu->setMedium(cn->getMedium());
206  volu->setShape(cn->getShapePointer());
207  Int_t n=cn->getNumPoints();
208  volu->createPoints(n);
209  for (Int_t k=0; k<n; k++) { volu->setPoint(k,*(cn->getPoint(k))); }
210  }
211  }
212 
213  } else {
214  // cbm format
215  Ssiz_t l=name.Last('#');
216  if (l>0) {
217  char c;
218  do {
219  c=fin.get();
220  } while (c==' ' || c=='\n');
221  Int_t i=(Int_t)c;
222  fin.putback(c);
223  if (!isalpha(i)) { isCopy=kTRUE; }
224  refName=name(0,l);
225  refNode=(FairGeoCopyNode*)refVolumes->FindObject(refName);
226  if (!refNode) {
227  if (isCopy) { return kFALSE; }
228  refVolumes->Add(new FairGeoCopyNode(refName.Data(),volu));
229  } else {
230  FairGeoNode* cn=refNode->pNode;
231  volu->setCopyNode(cn);
232  volu->setMedium(cn->getMedium());
233  volu->setShape(cn->getShapePointer());
234  Int_t n=cn->getNumPoints();
235  volu->createPoints(n);
236  for (Int_t k=0; k<n; k++) { volu->setPoint(k,*(cn->getPoint(k))); }
237  }
238  }
239 
240  }
241 
242  if (!isCopy) {
243  fin>>buf;
244  FairGeoBasicShape* sh=pShapes->selectShape(buf);
245  if (sh) { volu->setShape(sh); }
246  else { return kFALSE; }
247  fin>>buf;
248  FairGeoMedium* medium=media->getMedium(buf);
249  if (!medium) {
250  Error("readVolumeParams","Medium %s not found in list of media",buf);
251  return kFALSE;
252  }
253  volu->setMedium(medium);
254  Int_t n=0;
255  fin.getline(buf,maxbuf); // to read the end of line
256  if (sh) { n=sh->readPoints(&fin,volu); }
257  if (n<=0) { return kFALSE; }
258  }
259  readTransform(fin,volu->getTransform());
260  return kTRUE;
261 }

263 Bool_t FairGeoSet::readKeepIn(fstream& fin,FairGeoMedia* media,TString& name)
264 {
265  // Reads the keepin volume from file
266  fin.clear();
267  fin.seekg(0, ios::beg);
268  FairGeoNode* volu=0;
269  Bool_t rc=kTRUE;
270  TList refVolumes;
271  const Int_t maxbuf=256;
272  char buf[maxbuf];
273  while(rc && !volu && !fin.eof()) {
274  fin>>buf;
275  if (buf[0]=='/') { fin.getline(buf,maxbuf); }
276  else if (!fin.eof()) {
277  if (strcmp(buf,name)==0) {
278  volu=new FairGeoNode;
279  volu->SetName(buf);
280  volu->setVolumeType(kFairGeoKeepin);
281  readInout(fin);
282  rc=readVolumeParams(fin,media,volu,&refVolumes);
283  } else {
284  do {
285  fin.getline(buf,maxbuf);
286  } while(!fin.eof()&&buf[0]!='/');
287  }
288  }
289  }
290  if (rc) { volumes->Add(volu); }
291  else {
292  delete volu;
293  volu=0;
294  }
295  refVolumes.Delete();
296  return rc;
297 }

299 Bool_t FairGeoSet::readModule(fstream& fin,FairGeoMedia* media,TString& modName,
300  TString& eleName,Bool_t containsActiveMod)
301 {
302  // Reads the whole geometry of a module from file
303  const Int_t maxbuf=256;
304  Text_t buf[maxbuf];
305  fin.clear();
306  fin.seekg(0,ios::beg);
307  FairGeoNode* volu=0;
308  Bool_t rc=kTRUE;
309  TList refVolumes;
310  TString name;
311  while(rc && !fin.eof()) {
312  fin>>buf;
313  if (buf[0]=='/') { fin.getline(buf,maxbuf); }
314  else if (!fin.eof()) {
315  //TString name(buf);
316  name=buf;
317  if (name.BeginsWith(modName)) {
318  volu=new FairGeoNode;
319  volu->SetName(buf);
320  volu->setVolumeType(kFairGeoModule);
321  Int_t a=getModule(getSecNumInMod(buf),getModNumInMod(buf));
322  if (a>0) { volu->setActive(kTRUE); }
323  else { volu->setActive(kFALSE); }
324  readInout(fin);
325  rc=readVolumeParams(fin,media,volu,&refVolumes);
326  if (rc) { volumes->Add(volu); }
327  else {
328  Error("readModule","Failed for module %s\n",volu->GetName());
329  delete volu;
330  volu=0;
331  }
332  } else if (!eleName.IsNull()&&name.BeginsWith(eleName)) {
333  volu=new FairGeoNode;
334  volu->SetName(buf);
335  volu->setVolumeType(kFairGeoElement);
336  volu->setActive(containsActiveMod);
337  rc=readVolumeParams(fin,media,volu,&refVolumes);
338  if (rc) { volumes->Add(volu); }
339  else {
340  Error("readModule","Failed for %s\n",volu->GetName());
341  delete volu;
342  volu=0;
343  }
344  } else {
345  do {
346  fin.getline(buf,maxbuf);
347  } while(!fin.eof()&&buf[0]!='/');
348  }
349  }
350  }
351  refVolumes.Delete();
352  return rc;
353 }
