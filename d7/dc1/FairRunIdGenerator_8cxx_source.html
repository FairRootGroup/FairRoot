58 int FairRunIdGenerator::get_random_fd(void)
59 {
60  struct timeval tv;
61  static int fd = -2;
62  int i;
63 
64  if (fd == -2) {
65  gettimeofday(&tv, 0);
66  fd = open("/dev/urandom", O_RDONLY);
67  if (fd == -1) {
68  fd = open("/dev/random", O_RDONLY | O_NONBLOCK);
69  }
70  srand((getpid() << 16) ^ getuid() ^ tv.tv_sec ^ tv.tv_usec);
71  }
72  /* Crank the random number generator a few times */
73  gettimeofday(&tv, 0);
74  for (i = (tv.tv_sec ^ tv.tv_usec) & 0x1F; i > 0; i--) {
75  rand();
76  }
77  return fd;
78 }

81 /*
82  * Generate a series of random bytes. Use /dev/urandom if possible,
83  * and if not, use srandom/random.
84  */
85 void FairRunIdGenerator::get_random_bytes(void* buf, int nbytes)
86 {
87  int i, fd = get_random_fd();
88  int lose_counter = 0;
89  char* cp = (char*) buf;
90 
91  if (fd >= 0) {
92  while (nbytes > 0) {
93  i = read(fd, cp, nbytes);
94  if ((i < 0) &&
95  ((errno == EINTR) || (errno == EAGAIN))) {
96  continue;
97  }
98  if (i <= 0) {
99  if (lose_counter++ == 8) {
100  break;
101  }
102  continue;
103  }
104  nbytes -= i;
105  cp += i;
106  lose_counter = 0;
107  }
108  }
109  /* XXX put something better here if no /dev/random! */
110  for (i = 0; i < nbytes; i++) {
111  *cp++ = rand() & 0xFF;
112  }
113  return;
114 }

119 int FairRunIdGenerator::get_node_id(unsigned char* node_id)
120 {
121 #ifdef HAVE_NET_IF_H
122  int sd;
123  struct ifreq ifr, *ifrp;
124  struct ifconf ifc;
125  char buf[1024];
126  int n, i;
127  unsigned char* a;
128 
129  /*
130  * BSD 4.4 defines the size of an ifreq to be
131  * max(sizeof(ifreq), sizeof(ifreq.ifr_name)+ifreq.ifr_addr.sa_len
132  * However, under earlier systems, sa_len isn't present, so the size is
133  * just sizeof(struct ifreq)
134  */
135 #ifdef HAVE_SA_LEN
136 #ifndef max
137 #define max(a,b) ((a) > (b) ? (a) : (b))
138 #endif
139 #define ifreq_size(i) max(sizeof(struct ifreq),\
140  sizeof((i).ifr_name)+(i).ifr_addr.sa_len)
141 #else
142 #define ifreq_size(i) sizeof(struct ifreq)
143 #endif /* HAVE_SA_LEN*/
144 
145  sd = socket(AF_INET, SOCK_DGRAM, IPPROTO_IP);
146  if (sd < 0) {
147  return -1;
148  }
149  memset(buf, 0, sizeof(buf));
150  ifc.ifc_len = sizeof(buf);
151  ifc.ifc_buf = buf;
152  if (ioctl(sd, SIOCGIFCONF, (char*) &ifc) < 0) {
153  close(sd);
154  return -1;
155  }
156  n = ifc.ifc_len;
157  for (i = 0; i < n; i += ifreq_size(*ifr)) {
158  ifrp = (struct ifreq*) ((char*) ifc.ifc_buf + i);
159  strncpy(ifr.ifr_name, ifrp->ifr_name, IFNAMSIZ);
160 #ifdef SIOCGIFHWADDR
161  if (ioctl(sd, SIOCGIFHWADDR, &ifr) < 0) {
162  continue;
163  }
164  a = (unsigned char*) &ifr.ifr_hwaddr.sa_data;
165 #else
166 #ifdef SIOCGENADDR
167  if (ioctl(sd, SIOCGENADDR, &ifr) < 0) {
168  continue;
169  }
170  a = (unsigned char*) ifr.ifr_enaddr;
171 #else
172  /*
173  * XXX we don't have a way of getting the hardware
174  * address
175  */
176  close(sd);
177  return 0;
178 #endif /* SIOCGENADDR */
179 #endif /* SIOCGIFHWADDR */
180  if (!a[0] && !a[1] && !a[2] && !a[3] && !a[4] && !a[5]) {
181  continue;
182  }
183  if (node_id) {
184  memcpy(node_id, a, 6);
185  close(sd);
186  return 1;
187  }
188  }
189  close(sd);
190 #endif
191  return 0;
192 }

197 int FairRunIdGenerator::get_clock(uint32_t* clock_high, uint32_t* clock_low, uint16_t* ret_clock_seq)
198 {
199  static int adjustment = 0;
200  static struct timeval last =
201  {0, 0};
202  static uint16_t clock_seq;
203  struct timeval tv;
204  unsigned long long clock_reg;
205 
206 try_again:
207  gettimeofday(&tv, 0);
208  if ((last.tv_sec == 0) && (last.tv_usec == 0)) {
209  get_random_bytes(&clock_seq, sizeof(clock_seq));
210  clock_seq &= 0x1FFF;
211  last = tv;
212  last.tv_sec--;
213  }
214  if ((tv.tv_sec < last.tv_sec) ||
215  ((tv.tv_sec == last.tv_sec) &&
216  (tv.tv_usec < last.tv_usec))) {
217  clock_seq = (clock_seq + 1) & 0x1FFF;
218  adjustment = 0;
219  last = tv;
220  } else if ((tv.tv_sec == last.tv_sec) &&
221  (tv.tv_usec == last.tv_usec)) {
222  if (adjustment >= MAX_ADJUSTMENT) {
223  goto try_again;
224  }
225  adjustment++;
226  } else {
227  adjustment = 0;
228  last = tv;
229  }
230 
231  clock_reg = tv.tv_usec * 10 + adjustment;
232  clock_reg += ((unsigned long long) tv.tv_sec) * 10000000;
233  clock_reg += (((unsigned long long) 0x01B21DD2) << 32) + 0x13814000;
234 
235  *clock_high = clock_reg >> 32;
236  *clock_low = clock_reg;
237  *ret_clock_seq = clock_seq;
238 
239  fTimeSpec.tv_sec = last.tv_sec;
240  // fTimeSpec.tv_nsec = last.tv_usec*1000.;
241  fTimeSpec.tv_nsec = 0.;
242 
243  return 0;
244 }

246 void FairRunIdGenerator::uuid_generate_time(uuid_t out)
247 {
248  static unsigned char node_id[6];
249  static int has_init = 0;
250  struct uuid uu;
251  uint32_t clock_mid;
252 
253  if (!has_init) {
254  if (get_node_id(node_id) <= 0) {
255  get_random_bytes(node_id, 6);
256  /*
257  * Set multicast bit, to prevent conflicts
258  * with IEEE 802 addresses obtained from
259  * network cards
260  */
261  node_id[0] |= 0x80;
262  }
263  has_init = 1;
264  }
265  get_clock(&clock_mid, &uu.time_low, &uu.clock_seq);
266  uu.clock_seq |= 0x8000;
267  uu.time_mid = (uint16_t) clock_mid;
268  uu.time_hi_and_version = (clock_mid >> 16) | 0x1000;
269  memcpy(uu.node, node_id, 6);
270  uuid_pack(&uu, out);
271 }
