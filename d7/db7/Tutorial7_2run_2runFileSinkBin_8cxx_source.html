170 int main(int argc, char** argv)
171 {
172  s_catch_signals();
173  DeviceOptions_t options;
174  try
175  {
176  if (!parse_cmd_line(argc, argv, &options))
177  return 0;
178  }
179  catch (exception& e)
180  {
181  MQLOG(ERROR) << e.what();
182  return 1;
183  }
184 
185  MQLOG(INFO) << "PID: " << getpid();
186 
187 #ifdef NANOMSG
188  FairMQTransportFactory* transportFactory = new FairMQTransportFactoryNN();
189 #else
190  FairMQTransportFactory* transportFactory = new FairMQTransportFactoryZMQ();
191 #endif
192 
193  filesink.SetTransport(transportFactory);
194 
195  filesink.SetProperty(TSink::Id, options.id);
196  filesink.SetProperty(TSink::NumIoThreads, options.ioThreads);
197 
198  filesink.SetProperty(TSink::NumInputs, 1);
199  filesink.SetProperty(TSink::NumOutputs, 0);
200 
201  filesink.InitInputContainer( options.classname.c_str() );
202  filesink.SetFileProperties(options.filename,options.treename,options.branchname,options.classname,
203  options.fileoption,options.useTClonesArray);
204 
205 
206  filesink.ChangeState(TSink::INIT);
207  filesink.SetProperty(TSink::InputSocketType, options.inputSocketType);
208  filesink.SetProperty(TSink::InputRcvBufSize, options.inputBufSize);
209  filesink.SetProperty(TSink::InputMethod, options.inputMethod);
210  filesink.SetProperty(TSink::InputAddress, options.inputAddress);
211 
212  filesink.ChangeState(TSink::SETOUTPUT);
213  filesink.ChangeState(TSink::SETINPUT);
214  filesink.ChangeState(TSink::BIND);
215  filesink.ChangeState(TSink::CONNECT);
216  filesink.ChangeState(TSink::RUN);
217 
218  // wait until the running thread has finished processing.
219  boost::unique_lock<boost::mutex> lock(filesink.fRunningMutex);
220  while (!filesink.fRunningFinished)
221  {
222  filesink.fRunningCondition.wait(lock);
223  }
224 
225  filesink.ChangeState(TSink::STOP);
226  filesink.ChangeState(TSink::END);
227 
228  return 0;
229 }
