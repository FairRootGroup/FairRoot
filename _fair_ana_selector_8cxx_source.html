41  // Init() will be called many times when running on PROOF

59

61

83  LOG(INFO) << "FairAnaSelector::Init(): out status : \"" << outputStat.Data() << "\"" << FairLogger::endl;

89  outFileName.Remove(outFileName.Length()-5);

107 
108  // ----- Parameter database --------------------------------------------
109  FairRuntimeDb* rtdb = fRunAna->GetRuntimeDb();
110 
111  if ( par1Str.Contains(".root") ) {
112  FairParRootFileIo* parInput1 = new FairParRootFileIo();

114  rtdb->setFirstInput (parInput1);

191 Bool_t FairAnaSelector::Process(Long64_t entry)
192 {
193  // The Process() function is called for each entry in the tree (or possibly
194  // keyed object in the case of PROOF) to be processed. The entry argument
195  // specifies which entry in the currently loaded tree is to be processed.
196  // It can be passed to either FairAnaSelector::GetEntry() or TBranch::GetEntry()
197  // to read either all or the required parts of the data. When processing
198  // keyed objects with PROOF, the object is already loaded and is available
199  // via the fObject pointer.
200  //
201  // This function should contain the "body" of the analysis. It can contain
202  // simple or elaborate selection criteria, run algorithms on the data
203  // of the event and typically fill histograms.
204  //
205  // The processing can be stopped by calling Abort().
206  //
207  // Use fStatus to set the return value of TTree::Process().
208  //
209  // The return value is currently not used.
210  // LOG(INFO) << "FairAnaSelector::Process(): Proceeding to analyze event " << entry << "." << FairLogger::endl;
211 
212  fRunAna->RunOneEvent(entry);
213 
214  // LOG(INFO) << "FairAnaSelector::Process(): Event " << entry << " analyzed." << FairLogger::endl;
215  return kTRUE;
216 }
