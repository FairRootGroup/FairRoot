166 // ----- Public method GenerateEvent -----------------------------------
167 Bool_t FairPrimaryGenerator::GenerateEvent(FairGenericStack *pStack) {
168  // Check for MCEventHeader
169  if (!fEvent) {
170  LOG(FATAL) << "No MCEventHeader branch!" << FairLogger::endl;
171  return kFALSE;
172  } else {
173 
174  // Initialise
175  fStack = pStack;
176  fNTracks = 0;
177  fEvent->Reset();
178 
179  // Create event vertex
180  MakeVertex();
181  fEvent->SetVertex(fVertex);
182 
183  // Create beam angle
184  // Here we only randomly generate two angles (anglex, angley)
185  // for the event and later on (in AddTrack())
186  // all our particles will be rotated accordingly.
187  if (fBeamAngle) {
188  MakeBeamAngle();
189  }
190 
191  // Create event plane
192  // Randomly generate an angle by which each track added (in AddTrack())
193  // to the event is rotated around the z-axis
194  if (fEventPlane) {
195  MakeEventPlane();
196  }
197 
198  // Call the ReadEvent methods from all registered generators
199  fListIter->Reset();
200  TObject *obj = 0;
201  FairGenerator *gen = 0;
202  while ((obj = fListIter->Next())) {
203  gen = dynamic_cast<FairGenerator *>(obj);
204  if (!gen) {
205  return kFALSE;
206  }
207  const char *genName = gen->GetName();
208  fMCIndexOffset = fNTracks; // number tracks before generator is called
209  Bool_t test = gen->ReadEvent(this);
210  if (!test) {
211  LOG(ERROR) << "ReadEvent failed for generator " << genName
212  << FairLogger::endl;
213  return kFALSE;
214  }
215  }
216 
217 
218  fTotPrim += fNTracks;
219  // Screen output
220 
221  // Set the event number if not set already by one of the dedicated generators
222  if (-1 == fEvent->GetEventID()) {
223  fEventNr++;
224  fEvent->SetEventID(fEventNr);
225  }
226  LOG(DEBUG) << "(Event " << fEvent->GetEventID() << ") " << fNTracks
227  << " primary tracks from vertex (" << fVertex.X() << ", "
228  << fVertex.Y() << ", " << fVertex.Z() << ") with beam angle ("
229  << fBeamAngleX << ", " << fBeamAngleY << ") " << FairLogger::endl;
230 
231  fEvent->SetNPrim(fNTracks);
232 
233  return kTRUE;
234  }
235 }

238 // ----- Public method AddTrack ----------------------------------------
239 void FairPrimaryGenerator::AddTrack(Int_t pdgid, Double_t px_raw,
240  Double_t py_raw, Double_t pz_raw,
241  Double_t vx, Double_t vy, Double_t vz,
242  Int_t parent, Bool_t wanttracking,
243  Double_t e, Double_t tof, Double_t weight) {
244 
245  // ---> Add event vertex to track vertex
246  vx += fVertex.X();
247  vy += fVertex.Y();
248  vz += fVertex.Z();
249 
250  // cout << "FairPrimaryGenerator::AddTrack +Z" << fVertex.Z() << " pdgid " <<
251  // pdgid << "?"<< wanttracking << endl;
252 
253  TVector3 mom(px_raw, py_raw, pz_raw);
254 
255  if (fEventPlane) {
256  // Rotate the track (event) by the randomly generated angle which is fixed
257  // for the complete event.
258  // The coordinate system is not changed by this rotation.
259  mom.RotateZ(fPhi);
260  }
261 
262  if (fBeamAngle) {
263  // Rotate the track (event) from the rotated beam direction system into
264  // the fixed global experiment coordinate system
265  mom.RotateUz(fBeamDirection.Unit());
266  }
267 
268  // ---> Convert K0 and AntiK0 into K0s and K0L
269  if (pdgid == 311 || pdgid == -311) {
270  Double_t test = gRandom->Uniform(0., 1.);
271  if (test >= 0.5) {
272  pdgid = 310;
273  } // K0S
274  else {
275  pdgid = 130;
276  } // K0L
277  }
278 
279  // ---> Check whether particle type is in PDG Database
280  TDatabasePDG *pdgBase = TDatabasePDG::Instance();
281  if (!pdgBase) {
282  Fatal("FairPrimaryGenerator", "No TDatabasePDG instantiated");
283  } else {
284  TParticlePDG *pdgPart = pdgBase->GetParticle(pdgid);
285  if (!pdgPart) {
286  if (e < 0) {
287  cerr << "\033[5m\033[31m -E FairPrimaryGenerator: PDG code " << pdgid
288  << " not found in database.\033[0m " << endl;
289  cerr << "\033[5m\033[31m -E FairPrimaryGenerator: Discarding particle "
290  "\033[0m " << endl;
291  cerr << "\033[5m\033[31m -E FairPrimaryGenerator: now MC Index is "
292  "corrupted \033[0m " << endl;
293  return;
294  } else {
295  cout << "\033[5m\033[31m -W FairPrimaryGenerator: PDG code " << pdgid
296  << " not found in database. This warning can be savely "
297  "ignored.\033[0m " << endl;
298  }
299  }
300  }
301  // ---> Get mass and calculate energy of particle
302  if (e < 0) {
303  Double_t mass = pdgBase->GetParticle(pdgid)->Mass();
304  e = TMath::Sqrt(mom.Mag2() + mass * mass);
305  } // else, use the value of e given to the function
306 
307  // ---> Set all other parameters required by PushTrack
308  Int_t doTracking = 0; // Go to tracking
309  if (fdoTracking && wanttracking) {
310  doTracking = 1;
311  }
312  Int_t dummyparent = -1; // Primary particle (now the value is -1 by default)
313  Double_t polx = 0.; // Polarisation
314  Double_t poly = 0.;
315  Double_t polz = 0.;
316  Int_t ntr = 0; // Track number; to be filled by the stack
317  Int_t status = 0; // Generation status
318 
319  if (parent != -1) {
320  parent += fMCIndexOffset;
321  } // correct for tracks which are in list before generator is called
322  // Add track to stack
323  fStack->PushTrack(doTracking, dummyparent, pdgid, mom.X(), mom.Y(), mom.Z(),
324  e, vx, vy, vz, tof, polx, poly, polz, kPPrimary, ntr,
325  weight, status, parent);
326  fNTracks++;
327 }

405 // ----- Private method MakeVertex -------------------------------------
406 void FairPrimaryGenerator::MakeVertex() {
407  Double_t vx = fBeamX0;
408  Double_t vy = fBeamY0;
409  Double_t vz;
410  if (1 == fNrTargets) {
411  vz = fTargetZ[0];
412  } else {
413  Int_t Target = static_cast<Int_t>(gRandom->Uniform(fNrTargets));
414  vz = fTargetZ[Target];
415  }
416 
417  if (fSmearVertexZ)
418  vz = gRandom->Uniform(vz - fTargetDz / 2., vz + fTargetDz / 2.);
419 
420  if (fSmearGausVertexZ) {
421  vz = gRandom->Gaus(vz, fTargetDz);
422  }
423 
424  if (fSmearGausVertexXY) {
425  if (fBeamSigmaX != 0.) {
426  vx = gRandom->Gaus(fBeamX0, fBeamSigmaX);
427  }
428  if (fBeamSigmaY != 0.) {
429  vy = gRandom->Gaus(fBeamY0, fBeamSigmaY);
430  }
431  }
432 
433  if (fSmearVertexXY) {
434  if (fBeamSigmaX != 0.) {
435  vx = gRandom->Uniform(vx - fBeamSigmaX / 2., vx + fBeamSigmaX / 2.);
436  }
437  if (fBeamSigmaY != 0.) {
438  vy = gRandom->Uniform(vy - fBeamSigmaY / 2., vy + fBeamSigmaY / 2.);
439  }
440  }
441 
442  fVertex = TVector3(vx, vy, vz);
443 }
