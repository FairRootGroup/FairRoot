24 FairMQSocketZMQ::FairMQSocketZMQ(const string& type, int num, int numIoThreads)
25  : FairMQSocket(ZMQ_SNDMORE, ZMQ_RCVMORE, ZMQ_DONTWAIT)
26  , fSocket(NULL)

28  , fBytesTx(0)
29  , fBytesRx(0)
30  , fMessagesTx(0)

32 {
33  stringstream id;
34  id << type << "." << num;
35  fId = id.str();
36 
37  int rc = zmq_ctx_set(fContext->GetContext(), ZMQ_IO_THREADS, numIoThreads);
38  if (rc != 0)
39  {
40  LOG(ERROR) << "failed configuring context, reason: " << zmq_strerror(errno);
41  }
42 
43  fSocket = zmq_socket(fContext->GetContext(), GetConstant(type));
44 
45  if (fSocket == NULL)
46  {
47  LOG(ERROR) << "failed creating socket #" << fId << ", reason: " << zmq_strerror(errno);
48  exit(EXIT_FAILURE);
49  }
50 
51  rc = zmq_setsockopt(fSocket, ZMQ_IDENTITY, &fId, fId.length());
52  if (rc != 0)
53  {
54  LOG(ERROR) << "failed setting ZMQ_IDENTITY socket option, reason: " << zmq_strerror(errno);
55  }
56 
57  // Tell socket to try and send/receive outstanding messages for <linger> milliseconds before terminating.
58  // Default value for ZeroMQ is -1, which is to wait forever.
59  int linger = 500;
60  rc = zmq_setsockopt(fSocket, ZMQ_LINGER, &linger, sizeof(linger));
61  if (rc != 0)
62  {
63  LOG(ERROR) << "failed setting ZMQ_LINGER socket option, reason: " << zmq_strerror(errno);
64  }
65 
66  if (type == "sub")
67  {
68  rc = zmq_setsockopt(fSocket, ZMQ_SUBSCRIBE, NULL, 0);
69  if (rc != 0)
70  {
71  LOG(ERROR) << "failed setting ZMQ_SUBSCRIBE socket option, reason: " << zmq_strerror(errno);
72  }
73  }
74 
75  LOG(INFO) << "created socket #" << fId;
76 }
