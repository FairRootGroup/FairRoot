162  bool CheckVerboseOutput(std::string logLevel, std::string outputString,
163  std::string verboseLevel, std::string resultString
164  ) {
165  // Example Output for logging level INFO and different verbosity levels
166  // verbosity Level:Output
167  // LOW :[INFO ] I am here.
168  // MEDIUM :[INFO ] [_TestFairTools.cxx::LogNoArguments:78] I am here.
169  // HIGH :[INFO ] [05.12.2011 12:00:10] [_TestFairTools.cxx::LogNoArguments:78] I am here.
170  std::string regexString = "\\[" + logLevel + "[ ]*][ ]";
171  if (verboseLevel.compare("HIGH") == 0) {
172  regexString = regexString + "\\[\\d{2}\\.\\d{2}\\.\\d{4}[ ]\\d{2}:\\d{2}:\\d{2}][ ]";
173  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
174  }
175  if (verboseLevel.compare("MEDIUM") == 0) {
176  regexString = regexString + "\\[.*::.*:\\d{2}][ ]";
177  }
178  regexString = regexString + outputString + ".*";
179 
180  const boost::regex e(regexString);
181  return regex_match(resultString, e);
182  }
