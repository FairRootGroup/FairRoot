156 void FairModule::ConstructASCIIGeometry(T dataType1, TString containerName, U dataType)
157 {
158  FairGeoLoader* loader=FairGeoLoader::Instance();
159  FairGeoInterface* GeoInterface =loader->getGeoInterface();
160  T* MGeo=new T();
161  MGeo->print();
162  MGeo->setGeomFile(GetGeometryFileName());
163  GeoInterface->addGeoModule(MGeo);
164  Bool_t rc = GeoInterface->readSet(MGeo);
165  if ( rc ) { MGeo->create(loader->getGeoBuilder()); }
166 
167  TList* volList = MGeo->getListOfVolumes();
168  // store geo parameter
169  FairRun* fRun = FairRun::Instance();
170  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
171 
172  dataType1 = *MGeo;
173 
174  if ( "" != containerName) {
175  LOG(INFO) << "Add GeoNodes for "<< MGeo->getDescription()
176  << " to container " << containerName << FairLogger::endl;
177 
178  // U par=(U)(rtdb->getContainer(containerName));
179  U* par=(U*)(rtdb->getContainer(containerName));
180  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
181  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
182 
183  TListIter iter(volList);
184  FairGeoNode* node = NULL;
185  FairGeoVolume* aVol=NULL;
186 
187  while( (node = (FairGeoNode*)iter.Next()) ) {
188  aVol = dynamic_cast<FairGeoVolume*> ( node );
189  if ( node->isSensitive() ) {
190  fSensNodes->AddLast( aVol );
191  } else {
192  fPassNodes->AddLast( aVol );
193  }
194  }
195  ProcessNodes( volList );
196  par->setChanged();
197  par->setInputVersion(fRun->GetRunId(),1);
198 
199  }
200 }
