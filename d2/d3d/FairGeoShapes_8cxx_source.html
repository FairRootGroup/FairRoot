13 // FairGeoShapes
14 //
15 // Class to manage the GEANT/ROOT geometry shapes
16 //
17 // The 3D geometry in ROOT uses the ROOT library geom.
18 //
19 // This class holds a list of shape classes (type FairGeoBasicShape). They are
20 // used by all detectors when reading or writing geometry parameter containers
21 // to ascii files or to create an event display. This class is instantiated in
22 // the class HSpectrometer inside the function createDetGeomPar(Text_t* name).
23 // Every geometry parameter container gets a pointer to it.
24 //
25 // The individual shape classes are accessible with the selectShape(...)
26 // functions. The functions readPoints(...) and writePoints(...) use it
27 // internally.
28 //
29 // In the current version the following shapes are implemented:
30 // BOX TRAP TRD1 PGON PCON TUBE TUBS CONE CONS SPHE ELTU
31 // TORUS is added 06.11.06 m.al-turany
32 //

84 FairGeoBasicShape* FairGeoShapes::selectShape(const TString& name)
85 {
86  // returns a pointer to the shape given by name
87  // creates a shape object and adds it to the list of shapes if
88  // not existing
89  // returns NULL if the corresponding shape class is not implemented
90  TString allShapes[13]= {"BOX ","TRAP","TRD1","PGON","PCON","TUBE","TUBS",
91  "CONE","CONS","SPHE","ELTU","TORUS", "ASSEMBLY"
92  };
93  TString sName(name);
94  if (sName.Length()==3) { sName+=" "; }
95  FairGeoBasicShape* s=(FairGeoBasicShape*)shapes->FindObject(sName);
96  if (s) { return s; }
97  Int_t no=-1;
98  for(Int_t i=0; i<13; i++) {if (sName.CompareTo(allShapes[i])==0) { no=i; }}
99  switch(no) {
100  case 0:
101  {s= new FairGeoBrik(); break;}
102  case 1:
103  {s= new FairGeoTrap(); break;}
104  case 2:
105  {s= new FairGeoTrd1(); break;}
106  case 3:
107  {s= new FairGeoPgon(); break;}
108  case 4:
109  {s= new FairGeoPcon(); break;}
110  case 5:
111  {s= new FairGeoTube(); break;}
112  case 6:
113  {s= new FairGeoTubs(); break;}
114  case 7:
115  {s= new FairGeoCone(); break;}
116  case 8:
117  {s= new FairGeoCons(); break;}
118  case 9:
119  {s= new FairGeoSphe(); break;}
120  case 10:
121  {s= new FairGeoEltu(); break;}
122  case 11:
123  {s= new FairGeoTorus(); break;}
124  case 12:
125  {s= new FairGeoAssembly(); break;}
126  default: {
127  Error("selectShape","shape %s not implemented",name.Data());
128  }
129  }
130  if (s) { shapes->Add(s); }
131  return s;
132 }
