78 Bool_t FairTutorialDet1::ProcessHits(FairVolume* vol)
79 {
82  LOG(DEBUG) << "In FairTutorialDet1::ProcessHits" << FairLogger::endl;
83  //Set parameters at entrance of volume. Reset ELoss.
84  if ( TVirtualMC::GetMC()->IsTrackEntering() ) {
85  fELoss = 0.;
86  fTime = TVirtualMC::GetMC()->TrackTime() * 1.0e09;
87  fLength = TVirtualMC::GetMC()->TrackLength();
88  TVirtualMC::GetMC()->TrackPosition(fPos);
89  TVirtualMC::GetMC()->TrackMomentum(fMom);
90  }
91 
92  // Sum energy loss for all steps in the active volume
93  fELoss += TVirtualMC::GetMC()->Edep();
94 
95  // Create FairTutorialDet1Point at exit of active volume
96  if ( TVirtualMC::GetMC()->IsTrackExiting() ||
97  TVirtualMC::GetMC()->IsTrackStop() ||
98  TVirtualMC::GetMC()->IsTrackDisappeared() ) {
99  fTrackID = TVirtualMC::GetMC()->GetStack()->GetCurrentTrackNumber();
100  fVolumeID = vol->getMCid();
101  if (fELoss == 0. ) { return kFALSE; }
102  AddHit(fTrackID, fVolumeID, TVector3(fPos.X(), fPos.Y(), fPos.Z()),
103  TVector3(fMom.Px(), fMom.Py(), fMom.Pz()), fTime, fLength,
104  fELoss);
105 
106  // Increment number of tutorial det points in TParticle
107  FairStack* stack = static_cast<FairStack*>(TVirtualMC::GetMC()->GetStack());
108  stack->AddPoint(kTutDet);
109  }
110 
111  return kTRUE;
112 }

149 void FairTutorialDet1::ConstructGeometry()
150 {
155  FairGeoLoader* geoLoad = FairGeoLoader::Instance();
156  FairGeoInterface* geoFace = geoLoad->getGeoInterface();
157  FairTutorialDet1Geo* Geo = new FairTutorialDet1Geo();
158  Geo->setGeomFile(GetGeometryFileName());
159  geoFace->addGeoModule(Geo);
160 
161  Bool_t rc = geoFace->readSet(Geo);
162  if (rc) { Geo->create(geoLoad->getGeoBuilder()); }
163  TList* volList = Geo->getListOfVolumes();
164 
165  // store geo parameter
166  FairRun* fRun = FairRun::Instance();
167  FairRuntimeDb* rtdb= FairRun::Instance()->GetRuntimeDb();
168  FairTutorialDet1GeoPar* par=static_cast<FairTutorialDet1GeoPar*>(rtdb->getContainer("FairTutorialDet1GeoPar"));
169  TObjArray* fSensNodes = par->GetGeoSensitiveNodes();
170  TObjArray* fPassNodes = par->GetGeoPassiveNodes();
171 
172  TListIter iter(volList);
173  FairGeoNode* node = NULL;
174  FairGeoVolume* aVol=NULL;
175 
176  while( (node = static_cast<FairGeoNode*>(iter.Next())) ) {
177  aVol = dynamic_cast<FairGeoVolume*> ( node );
178  if ( node->isSensitive() ) {
179  fSensNodes->AddLast( aVol );
180  } else {
181  fPassNodes->AddLast( aVol );
182  }
183  }
184  par->setChanged();
185  par->setInputVersion(fRun->GetRunId(),1);
186 
187  ProcessNodes ( volList );
188 }
