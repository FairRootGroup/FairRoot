64 inline bool parse_cmd_line(int _argc, char* _argv[], DeviceOptions* _options)
65 {
66  if (_options == NULL)
67  throw std::runtime_error("Internal error: options' container is empty.");
68 
69  namespace bpo = boost::program_options;
70  bpo::options_description desc("Options");
71  desc.add_options()
72  ("id", bpo::value<string>()->required(), "Device ID")
73  ("io-threads", bpo::value<int>()->default_value(1), "Number of I/O threads")
74  ("input-socket-type", bpo::value<string>()->required(), "Input socket type: sub/pull")
75  ("input-buff-size", bpo::value<int>()->required(), "Input buffer size in number of messages (ZeroMQ)/bytes(nanomsg)")
76  ("input-method", bpo::value<string>()->required(), "Input method: bind/connect")
77  ("input-address", bpo::value<string>()->required(), "Input address, e.g.: \"tcp://*:5555\"")
78  ("help", "Print help messages");
79 
80  bpo::variables_map vm;
81  bpo::store(bpo::parse_command_line(_argc, _argv, desc), vm);
82 
83  if ( vm.count("help") )
84  {
85  LOG(INFO) << "FairMQ Bin Sink" << endl << desc;
86  return false;
87  }
88 
89  bpo::notify(vm);
90 
91  if ( vm.count("id") )
92  _options->id = vm["id"].as<string>();
93 
94  if ( vm.count("io-threads") )
95  _options->ioThreads = vm["io-threads"].as<int>();
96 
97  if ( vm.count("input-socket-type") )
98  _options->inputSocketType = vm["input-socket-type"].as<string>();
99 
100  if ( vm.count("input-buff-size") )
101  _options->inputBufSize = vm["input-buff-size"].as<int>();
102 
103  if ( vm.count("input-method") )
104  _options->inputMethod = vm["input-method"].as<string>();
105 
106  if ( vm.count("input-address") )
107  _options->inputAddress = vm["input-address"].as<string>();
108 
109  return true;
110 }

112 int main(int argc, char** argv)
113 {
114  s_catch_signals();
115 
116  DeviceOptions_t options;
117  try
118  {
119  if (!parse_cmd_line(argc, argv, &options))
120  return 0;
121  }
122  catch (exception& e)
123  {
124  LOG(ERROR) << e.what();
125  return 1;
126  }
127 
128  LOG(INFO) << "PID: " << getpid();

134 #endif

136  sink.SetTransport(transportFactory);
137 
138  sink.SetProperty(FairMQBinSink::Id, options.id);
139  sink.SetProperty(FairMQBinSink::NumIoThreads, options.ioThreads);
140 
141  sink.SetProperty(FairMQBinSink::NumInputs, 1);
142  sink.SetProperty(FairMQBinSink::NumOutputs, 0);
143 
144  sink.ChangeState(FairMQBinSink::INIT);
145 
146  sink.SetProperty(FairMQBinSink::InputSocketType, options.inputSocketType);
147  sink.SetProperty(FairMQBinSink::InputRcvBufSize, options.inputBufSize);
148  sink.SetProperty(FairMQBinSink::InputMethod, options.inputMethod);
149  sink.SetProperty(FairMQBinSink::InputAddress, options.inputAddress);
150 
151  sink.ChangeState(FairMQBinSink::SETOUTPUT);
152  sink.ChangeState(FairMQBinSink::SETINPUT);
153  sink.ChangeState(FairMQBinSink::RUN);
154 
155  // wait until the running thread has finished processing.
156  boost::unique_lock<boost::mutex> lock(sink.fRunningMutex);
157  while (!sink.fRunningFinished)
158  {
159  sink.fRunningCondition.wait(lock);
160  }
161 
162  sink.ChangeState(FairMQBinSink::STOP);
163  sink.ChangeState(FairMQBinSink::END);
164 
165  return 0;
166 }
