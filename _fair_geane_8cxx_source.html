67 InitStatus FairGeane::Init()

69 
70  cout << "------------------------FairGeane::Init ()------------------------------" << endl;

72  TString LibMacro;
73  TString LibFunction;
74  TString ConfigMacro;
75  TString cuts=fUserCuts;
76  if(cuts== TString("")) { cuts="SetCuts.C"; }

78  TString work = getenv("VMCWORKDIR");
79  TString work_config=work+"/gconfig/";
80  TString config_dir= getenv("CONFIG_DIR");
81  Bool_t AbsPath=kFALSE;
82  if (!config_dir.EndsWith("/")) { config_dir+="/"; }
83 
84 
85  TString geaneLibMacro="g3libs.C";
86  TString geaneMacro;
87  if(fUserConfig.IsNull()) {
88  geaneMacro="Geane.C";
89  fUserConfig = geaneMacro;
90  } else {
91  if (fUserConfig.Contains("/")) { AbsPath=kTRUE; }
92  geaneMacro = fUserConfig;
93  cout << "---------------User config is used : " << geaneMacro.Data() <<"-----------------"<< endl;
94  }
95  if (TString(gSystem->FindFile(config_dir.Data(),geaneLibMacro)) != TString("")) {
96  cout << "---User path for Configuration (g3libs.C) is used : " << config_dir.Data() << endl;
97  } else {
98  geaneLibMacro=work_config+"g3libs.C";
99  }
100  LibMacro=geaneLibMacro;
101  LibFunction="g3libs()";
102 
103  if (!AbsPath && TString(gSystem->FindFile(config_dir.Data(),geaneMacro)) != TString("")) {
104  cout << "---User path for Configuration (Geane.C) is used : " << config_dir.Data() << endl;
105  ConfigMacro =geaneMacro;
106  } else {
107  if(AbsPath) { ConfigMacro = fUserConfig; }
108  else { ConfigMacro =work_config+fUserConfig; }
109  }
110 
111  if (!AbsPath &&TString(gSystem->FindFile(config_dir.Data(),cuts)) != TString("")) {
112  cout << "---User path for Cuts and Processes (SetCuts.C) is used : " << config_dir.Data() << endl;
113  } else {
114  cuts =work_config+ "SetCuts.C";
115  }
116 
117  gROOT->LoadMacro(LibMacro.Data());
118  gROOT->ProcessLine(LibFunction.Data());
119 
120  gROOT->LoadMacro(ConfigMacro.Data());
121  gROOT->ProcessLine("Config()");
122 
123  //gROOT->LoadMacro(cuts);
124  //gROOT->ProcessLine("SetCuts()");

126  FairField* field=FairRunAna::Instance()->GetField();

128  fApp->SetField(field);
129 
130  fApp->InitMC(ConfigMacro.Data(), "");
131 
132  cout <<"I- FairGeane::FairGeane: Geane is Initialized " << endl;
