402 Bool_t FairGeoInterface::connectOutput (const char* name)
403 {
404  // Connects the output (ASCII or Oracle)
405  if (output) {
406  if (strcmp(output->IsA()->GetName(),"FairGeoAsciiIo")==0) {
407  TString fName(name);
408  char buf[80];
409  struct tm* newtime;
410  time_t t;
411  time(&t);
412  newtime=localtime(&t);
413  if (newtime->tm_mday<10) { sprintf(buf,"_0%i",newtime->tm_mday); }
414  else { sprintf(buf,"_%i",newtime->tm_mday); }
415  fName=fName+buf;
416  if (newtime->tm_mon<9) { sprintf(buf,"0%i",newtime->tm_mon+1); }
417  else { sprintf(buf,"%i",newtime->tm_mon+1); }
418  fName=fName+buf;
419  Int_t y=newtime->tm_year-100;
420  if (y<10) { sprintf(buf,"0%i",y); }
421  else { sprintf(buf,"%i",y); }
422  fName=fName+buf;
423  if (newtime->tm_hour<10) { sprintf(buf,"0%i",newtime->tm_hour); }
424  else { sprintf(buf,"%i",newtime->tm_hour); }
425  fName=fName+buf;
426  if (newtime->tm_min<10) { sprintf(buf,"0%i",newtime->tm_min); }
427  else { sprintf(buf,"%i",newtime->tm_min); }
428  fName=fName+buf;
429  if (newtime->tm_sec<10) { sprintf(buf,"0%i",newtime->tm_sec); }
430  else { sprintf(buf,"%i",newtime->tm_sec); }
431  fName=fName+buf+".geo";
432  output->open(fName,"out");
433  cout<<"Output file for "<<name<<": "
434  <<((FairGeoAsciiIo*)output)->getFilename()<<endl;
435  }
436  if (output->isOpen()&&output->isWritable()) { return kTRUE; }
437  } else { Error("connectOutput","No output open"); }
438  return kFALSE;
439 }
