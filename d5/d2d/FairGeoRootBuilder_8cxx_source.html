136 Int_t FairGeoRootBuilder::createMedium(FairGeoMedium* med)
137 {
138  // Creates the medium
139  if (!geoManager&&!med) { return 0; }
140  Int_t nComp=med->getNComponents();
141  Int_t weightFac=med->getWeightFac();
142  TGeoMaterial* material=0;
143  Double_t p[3];
144  if (nComp==1) {
145  med->getComponent(0,p);
146  material=new TGeoMaterial(med->GetName(),p[0],p[1],med->getDensity(),
147  med->getRadiationLength());
148  // Interaction length not defined!!!!!!
149  } else {
150  material=new TGeoMixture(med->GetName(),nComp,med->getDensity());
151  Double_t sumWeights=0.;
152  if (weightFac<0) {
153  for(Int_t i=0; i<nComp; i++) {
154  med->getComponent(i,p);
155  sumWeights+=p[0]*p[2];
156  }
157  }
158  for(Int_t i=0; i<nComp; i++) {
159  med->getComponent(i,p);
160  if (weightFac>0) {
161  ((TGeoMixture*)material)->DefineElement(i,p[0],p[1],p[2]);
162  } else {
163  ((TGeoMixture*)material)->DefineElement(i,p[0],p[1],p[0]*p[2]/sumWeights);
164  }
165  }
166  }
167  nMed++;
168  med->setMediumIndex(nMed);
169  Double_t mp[10];
170  med->getMediumPar(mp);
171  TGeoMedium* medium=new TGeoMedium(med->GetName(),nMed,material,mp);
172  if (medium) { return nMed; }
173  else { return 0; }
174 }
