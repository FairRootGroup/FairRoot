45 Bool_t FairMbsSource::Unpack(Int_t *data, Int_t size, Short_t type,
46  Short_t subType, Short_t procId, Short_t subCrate,
47  Short_t control) {
48  FairUnpack *unpack;
49  Bool_t seen = kFALSE;
50  for (Int_t i = 0; i < fUnpackers->GetEntriesFast(); i++) {
51  unpack = (FairUnpack *)fUnpackers->At(i);
52 
53  if (unpack->GetSubCrate() < 0) { // All sub-crates
54  if (type != unpack->GetType() || subType != unpack->GetSubType() ||
55  procId != unpack->GetProcId() || control != unpack->GetControl()) {
56  continue;
57  }
58  } else { // specified sub-crate
59  if (type != unpack->GetType() || subType != unpack->GetSubType() ||
60  procId != unpack->GetProcId() || subCrate != unpack->GetSubCrate() ||
61  control != unpack->GetControl()) {
62  continue;
63  }
64  }
65 
66  if (!unpack->DoUnpack(data, size)) {
67  return kFALSE;
68  }
69 
70  seen = kTRUE;
71  }
72  return seen;
73 }
