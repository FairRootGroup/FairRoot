53  if (fStopFunction == 0) { //no function is given ==> read in data in traditional way event by event

92  //if you are still at the end of the BufferArray than break (no new data in tree)

95  }

97  if (fVerbose > 1) {

99  }
100  }
101 
102  if (fVerbose > 1) {
103  std::cout << "-I- FairTSBufferFunctional::GetData Index for Absorb: " << posBuffer << " BufferArray size: " << fBufferArray->GetEntriesFast() << std::endl;
104  }

115  }
116 
117  if (posBuffer >= fBufferArray->GetEntriesFast() && posBuffer != 0 && fBranchIndex + 1 >= fBranch->GetEntries()) {

145 Int_t FairTSBufferFunctional::FindStartIndex(Double_t startParameter)
146 {
147  FairTimeStamp* dataPoint;
148  Int_t tempIndex = fBranchIndex;
149  Bool_t runBackwards = kTRUE;
150  Int_t previousIndex = -1;
151  Int_t previousBranchIndex = -1;
152 
153  ReadInEntry(tempIndex); //< Get Data out of Tree
154 
155  while (fInputArray->GetEntries() == 0 && tempIndex > 0) { //< If the entry of the tree was empty read in previous entries until you find one which is filled
156  tempIndex--;
157  ReadInEntry(tempIndex);
158  }
159 
160  if (fInputArray->GetEntries() == 0) { //If the previous entries in the tree are also empty run in the forward direction
161  ReadInNextFilledEntry();
162  runBackwards = kFALSE;
163  }
164 
165  if (fInputArray->GetEntries() == 0) { //If there is still no data the branch is empty!
166  std::cout << "-I- FairTSBufferFunctional::FindStartIndex: All entries are empty!" << std::endl;
167  return -1;
168  }
169  fBranchIndex = tempIndex;
170  //Now we have data or FindStartIndex already returned -1
171 
172  dataPoint = (FairTimeStamp*)fInputArray->Last();
173  //std::cout << "DataPoint: " << *dataPoint << std::endl;
174  while(!(*fStartFunction)(dataPoint, startParameter)) {
175  //std::cout << "DataPoint Search Entry: " << fBranchIndex << ": " << *dataPoint << std::endl;
176  ReadInNextFilledEntry();
177  if (fInputArray->GetEntries() != 0) {
178  dataPoint = (FairTimeStamp*)fInputArray->Last();
179  } else {
180  return -1;
181  }
182  }
183 
184  //Now you have data where the last element in the array does not fit to your request
185  Int_t startPos = fInputArray->GetEntries() - 1;
186  while ((*fStartFunction)(dataPoint, startParameter)) {
187  //std::cout << "DataPoint Search in Entry: " << fBranchIndex << ": " << *dataPoint << std::endl;
188  previousIndex = startPos;
189  previousBranchIndex = fBranchIndex;
190  startPos--;
191  if (startPos == -1) {
192  fBranchIndex = ReadInPreviousFilledEntry(fBranchIndex);
193  startPos = fInputArray->GetEntries() - 1;
194  if (startPos < 0) {
195  if (fBranchIndex == 0) {
196  return 0;
197  }
198  return -1;
199  }
200  }
201  dataPoint = (FairTimeStamp*)fInputArray->At(startPos);
202  }
203  fBranchIndex = previousBranchIndex;
204  return previousIndex;
205 }
