75 Bool_t FairContainer::setActualContext(const char* c)
76 {
77  // The function sets the actual context for the container, if it is in the list of
78  // accepted contexts. When the actual context was already set before, it prints a warning
79  // and ignores the second setting.
80  // The function returns kFALSE, when the context is not in the list.
81  if (contexts->FindObject(c)) {
82  if (actualContext.IsNull()) { actualContext=c; }
83  else Warning("addContext",
84  "Actual context of parameter container %s already defined as %s",
85  GetName(),actualContext.Data());
86  return kTRUE;
87  }
88  return kFALSE;

118 TString FairContainer::getConcatName()
119 {
120  // Returns the name of the parameter container used in the constructor and the
121  // runtime database.
122  // When the parameter container supportes different contexts (not only an empty string)
123  // and the actual context set is not the default context, the new name of the parameter
124  // container is concatinated as
125  // original container name + _ + actualcontext
126  TString cn=fName;
127  if (!actualContext.IsNull() && actualContext!=((TObjString*)contexts->At(0))->String()) {
128  cn+="_";
129  cn+=actualContext;
130  }
131  return cn;
132 }

171 FairParSet* FairContFact::getContainer(const char* name)
172 {
173  // Returns the pointer to the parameter container in the runtime database
174  // If this parameter container does not yet exit, it calls the function
175  // createContainer(FairContainer*), which is implemented in the derived classes
176  // and calls the corresponding constructor. Then the pointer it added in the
177  // runtime database.
178  FairContainer* c=(FairContainer*)(containers->FindObject(name));
179 
180  FairParSet* cont=0;
181  if (c) {
182  TString cn=c->getConcatName();
183  FairRuntimeDb* rtdb = FairRuntimeDb::instance();
184  if (!(cont=rtdb->findContainer(c->getConcatName().Data()))) {
185  if (strlen(c->getActualContext())==0) { c->setActualContext(c->getDefaultContext()); }
186  cont=createContainer(c);
187  if (cont) { rtdb->addContainer(cont); }
188  }
189  }
190  return cont;
191 }
