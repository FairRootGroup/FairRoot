14 // FairGeoTubs
15 //
16 // class for the GEANT shape TUBS
17 //
18 // The size of a TUBS is defined by 4 'points'.
19 // point 0: origin of starting circle of the tubs;
20 // point 1: inner radius of starting circle,
21 // outer radius of starting circle;
22 // (z-component not used)
23 // point 2: origin of ending circle of the tubs;
24 // point 3: starting angle of the segment,
25 // ending angle of the segment;
26 // (z-component not used)
27 // Warning: The x- and y-values of point 0 and 2 have to be the same!!!!
28 // A rotation has to be desribed via the rotation matrix.
29 //
30 // The intrinsic coordinate system of a TUBS, which sits in the CAVE and is
31 // not rotated, is identical with the laboratory system.
32 //

79 Int_t FairGeoTubs::readPoints(fstream* pFile,FairGeoVolume* volu)
80 {
81  // reads the 4 'points' decribed above from ascii file
82  // if the array of points is not existing in the volume it is created and
83  // the values are stored inside
84  // returns the number of points
85  if (!pFile) { return 0; }
86  if (volu->getNumPoints()!=nPoints) { volu->createPoints(nPoints); }
87  Double_t x,y,z;
88  const Int_t maxbuf=155;
89  Text_t buf[maxbuf];
90  for(Int_t i=0; i<nPoints; i++) {
91  pFile->getline(buf,maxbuf);
92  sscanf(buf,"%lf%lf%lf",&x,&y,&z);
93  volu->setPoint(i,x,y,z);
94  i++;
95  pFile->getline(buf,maxbuf);
96  sscanf(buf,"%lf%lf",&x,&y);
97  volu->setPoint(i,x,y,0.0);
98  }
99  return nPoints;
100 }

103 Bool_t FairGeoTubs::writePoints(fstream* pFile,FairGeoVolume* volu)
104 {
105  // writes the 4 'points' decribed above to ascii file
106  if (!pFile) { return kFALSE; }
107  Text_t buf[155];
108  for(Int_t i=0; i<nPoints; i++) {
109  FairGeoVector& v=*(volu->getPoint(i));
110  if (i==0 || i==2) { sprintf(buf,"%9.3f%10.3f%10.3f\n",v(0),v(1),v(2)); }
111  else { sprintf(buf,"%9.3f%10.3f\n",v(0),v(1)); }
112  pFile->write(buf,strlen(buf));
113  }
114  return kTRUE;
115 }

129 TArrayD* FairGeoTubs::calcVoluParam(FairGeoVolume* volu)
130 {
131  // calculates the parameters needed to create the shape TUBS
132  Double_t fac=10.;
133  FairGeoVector& v1=*(volu->getPoint(1));
134  param->AddAt(v1(0)/fac,0);
135  param->AddAt(v1(1)/fac,1);
136  FairGeoVector v=*(volu->getPoint(2)) - *(volu->getPoint(0));
137  param->AddAt(TMath::Abs(v(2))/fac/2.,2);
138  FairGeoVector& v3=*(volu->getPoint(3));
139  param->AddAt(v3(0),3);
140  param->AddAt(v3(1),4);
141  return param;
142 }
