# Create a library  

set(INCLUDE_DIRECTORIES
    ${CMAKE_SOURCE_DIR}/fairmq/logger
    ${CMAKE_SOURCE_DIR}/fairmq/logger/run
    )

include_directories(${INCLUDE_DIRECTORIES})
include_directories(${Boost_INCLUDE_DIRS})

set(LINK_DIRECTORIES ${Boost_LIBRARY_DIRS}) 

link_directories(${LINK_DIRECTORIES})

if(NOT ("${Boost_MAJOR_VERSION}.${Boost_MINOR_VERSION}" VERSION_LESS "1.56") )
    set(SRCS logger.cxx)
else()
    set(SRCS logger_oldboost_version.cxx)
endif()
set(LIBRARY_NAME fairmq_logger)

set(DEPENDENCIES
    boost_log
    boost_log_setup
    boost_thread
    boost_date_time
    boost_filesystem
    boost_system
    pthread
  )

install(FILES logger.h logger_def.h DESTINATION include/logger)

GENERATE_LIBRARY()

# generate test executable
set(EXE_NAME runtestLogger)
set(SRCS run/testLogger.cxx)
set(DEPENDENCIES fairmq_logger)
GENERATE_EXECUTABLE()
