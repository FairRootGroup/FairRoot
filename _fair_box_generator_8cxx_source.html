133 Bool_t FairBoxGenerator::Init()
134 {
135  // Initialize generator
136 
137 
138  if (fPhiMax-fPhiMin>360)
139  Fatal("Init()","FairBoxGenerator: phi range is too wide: %f<phi<%f",
140  fPhiMin,fPhiMax);
141  if (fPRangeIsSet && fPtRangeIsSet) {
142  Fatal("Init()","FairBoxGenerator: Cannot set P and Pt ranges simultaneously");
143  }
144  if (fPRangeIsSet && fYRangeIsSet) {
145  Fatal("Init()","FairBoxGenerator: Cannot set P and Y ranges simultaneously");
146  }
147  if ( (fThetaRangeIsSet && fYRangeIsSet) ||
148  (fThetaRangeIsSet && fEtaRangeIsSet) ||
149  (fYRangeIsSet && fEtaRangeIsSet) ) {
150  Fatal("Init()","FairBoxGenerator: Cannot set Y, Theta or Eta ranges simultaneously");
151  }
152  if (fPointVtxIsSet && fBoxVtxIsSet) {
153  Fatal("Init()","FairBoxGenerator: Cannot set point and box vertices simultaneously");
154  }
155 
156  // Check for particle type
157  TDatabasePDG* pdgBase = TDatabasePDG::Instance();
158  TParticlePDG* particle = pdgBase->GetParticle(fPDGType);
159 
160  if (! particle) {
161  Fatal("FairBoxGenerator","PDG code %d not defined.",fPDGType);
162  } else {
163  fPDGMass = particle->Mass();
164  // printf("particle->Mass() = %f \n", fPDGMass);
165  return kTRUE;
166  }
167 
168  return kTRUE;
169 }

172 Bool_t FairBoxGenerator::ReadEvent(FairPrimaryGenerator* primGen)
173 {
174  // Generate one event: produce primary particles emitted from one vertex.
175  // Primary particles are distributed uniformly along
176  // those kinematics variables which were limitted by setters.
177  // if SetCosTheta() function is used, the distribution will be uniform in
178  // cos(theta)
179 
180  Double32_t pabs=0, phi, pt=0, theta=0, eta, y, mt, px, py, pz=0;
181 
182  // Generate particles
183  for (Int_t k = 0; k < fMult; k++) {
184  phi = gRandom->Uniform(fPhiMin,fPhiMax) * TMath::DegToRad();
185 
186  if (fPRangeIsSet ) { pabs = gRandom->Uniform(fPMin,fPMax); }
187  else if (fPtRangeIsSet) { pt = gRandom->Uniform(fPtMin,fPtMax); }
188 
189  if (fThetaRangeIsSet) {
190  if (fCosThetaIsSet)
191  theta = acos(gRandom->Uniform(cos(fThetaMin* TMath::DegToRad()),
192  cos(fThetaMax* TMath::DegToRad())));
193  else {
194  theta = gRandom->Uniform(fThetaMin,fThetaMax) * TMath::DegToRad();
195  }
196  } else if (fEtaRangeIsSet) {
197  eta = gRandom->Uniform(fEtaMin,fEtaMax);
198  theta = 2*TMath::ATan(TMath::Exp(-eta));
199  } else if (fYRangeIsSet) {
200  y = gRandom->Uniform(fYMin,fYMax);
201  mt = TMath::Sqrt(fPDGMass*fPDGMass + pt*pt);
202  pz = mt * TMath::SinH(y);
203  }
204 
205  if (fThetaRangeIsSet || fEtaRangeIsSet) {
206  if (fPRangeIsSet ) {
207  pz = pabs*TMath::Cos(theta);
208  pt = pabs*TMath::Sin(theta);
209  } else if (fPtRangeIsSet) {
210  pz = pt/TMath::Tan(theta);
211  }
212  }
213 
214  px = pt*TMath::Cos(phi);
215  py = pt*TMath::Sin(phi);
216 
217  if (fBoxVtxIsSet) {
218  fX = gRandom->Uniform(fX1,fX2);
219  fY = gRandom->Uniform(fY1,fY2);
220  }
221 
222  if (fDebug)
223  printf("BoxGen: kf=%d, p=(%.2f, %.2f, %.2f) GeV, x=(%.1f, %.1f, %.1f) cm\n",
224  fPDGType, px, py, pz, fX, fY, fZ);
225 
226  primGen->AddTrack(fPDGType, px, py, pz, fX, fY, fZ);
227  }
228  return kTRUE;
229 
230 }
