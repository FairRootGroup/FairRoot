294 void ValTimeStamp::Set()
295 {
296 
297 #ifdef R__WIN32
298  ULARGE_INTEGER time;
299  GetSystemTimeAsFileTime((FILETIME*)&time);
300  fNanoSec = Int_t((time.QuadPart * (unsigned __int64) 100) %
301  (unsigned __int64) 1000000000);
302  time.QuadPart -=
303  (unsigned __int64) (1000*1000*10) // seconds
304  * (unsigned __int64) (60 * 60 * 24) // days
305  * (unsigned __int64) (134774); // # of days
306 
307  fSec = Int_t(time.QuadPart/(unsigned __int64) (1000*1000*10));
308 #else
309  struct timeval now;
310  if (!gettimeofday(&now,0)) {
311  fSec = now.tv_sec;
312  fNanoSec = now.tv_usec * 1000;
313  } else {
314  time_t nowtime;
315  time(&nowtime);
316  fSec = nowtime;
317  fNanoSec = 0;
318  }
319 #endif
320  static Int_t sec = 0, nsec = 0, fake_ns = 0;
321 
322  if (fSec == sec && fNanoSec == nsec) {
323  fNanoSec += ++fake_ns;
324  } else {
325  fake_ns = 0;
326  sec = fSec;
327  nsec = fNanoSec;
328  }

333 void ValTimeStamp::Set(Int_t year, Int_t month, Int_t day,
334  Int_t hour, Int_t min, Int_t sec,
335  Int_t nsec, Bool_t isUTC, Int_t secOffset)
336 {
337  if (year <= 37) { year += 2000; }
338  if (year >= 70 && year <= 137) { year += 1900; }
339  if (year >= 1900) { year -= 1900; }
340 
341  struct tm tmstruct;
342  tmstruct.tm_year = year;
343  tmstruct.tm_mon = month-1;
344  tmstruct.tm_mday = day;
345  tmstruct.tm_hour = hour;
346  tmstruct.tm_min = min;
347  tmstruct.tm_sec = sec + secOffset;
348  tmstruct.tm_isdst = -1;
349 
350  const time_t bad_time_t = (time_t) -1;
351  time_t utc_sec = (isUTC) ? MktimeFromUTC(&tmstruct) : mktime(&tmstruct);
352 
353  if (utc_sec == bad_time_t) {
354  cout << "-I- ValTimeStamp::Set time returned -1" << endl;
355  }
356 
357  fSec = utc_sec;
358  fNanoSec = nsec;
359 
360  NormalizeNanoSec();
361 }

404 time_t ValTimeStamp::MktimeFromUTC(tm_t* tmstruct)
405 {
406  const Int_t days[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
407  const Int_t daysLeap[] = {31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
408 
409  Int_t year = tmstruct->tm_year + 1900;
410  Bool_t isleap = ValTimeStamp::IsLeapYear(year);
411 
412  const Int_t* daysInMonth = days;
413  if (isleap) { daysInMonth = daysLeap; }
414 
415 
416  int& ref_tm_mon = tmstruct->tm_mon;
417  int& ref_tm_mday = tmstruct->tm_mday;
418  // count days in months past
419  tmstruct->tm_yday = 0;
420  for (Int_t imonth = 0; imonth < ref_tm_mon; imonth++) {
421  tmstruct->tm_yday += daysInMonth[imonth];
422  }
423  tmstruct->tm_yday += ref_tm_mday - 1;
424 
425 
426  while (ref_tm_mday > daysInMonth[ref_tm_mon]) {
427  ref_tm_mday -= daysInMonth[ref_tm_mon];
428  ref_tm_mon++;
429  }
430 
431 
432  tmstruct->tm_isdst = 0;
433 
434 
435  Int_t utc_sec = tmstruct->tm_sec +
436  tmstruct->tm_min*60 +
437  tmstruct->tm_hour*3600 +
438  tmstruct->tm_yday*86400 +
439  (tmstruct->tm_year-70)*31536000 +
440  ((tmstruct->tm_year-69)/4)*86400;
441 
442  return utc_sec;
443 }
