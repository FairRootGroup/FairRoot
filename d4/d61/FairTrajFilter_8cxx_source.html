106 Bool_t FairTrajFilter::IsAccepted(const TParticle* p) const
107 {
108  if( NULL == p ) {
109  return kFALSE;
110  }
111 
112  // Apply vertex cut
113  if( (p->Vx()<fVxMin) || (p->Vx()>fVxMax) ||
114  (p->Vy()<fVyMin) || (p->Vy()>fVyMax) ||
115  (p->Vz()<fVzMin) || (p->Vz()>fVzMax) ) {
116  return kFALSE;
117  }
118 
119  // Apply cut on kinematics
120  if( 0 == fKinCutType ) {
121  if( (p->P()<fPMin) || (p->P()>fPMax) ||
122  (p->Theta()<fThetaMin) || (p->Theta()>fThetaMax) ||
123  (p->Phi()<fPhiMin) || (p->Phi()>fPhiMax) ) {
124  return kFALSE;
125  }
126  } else if( 1 == fKinCutType ) {
127  if( (p->Px()<fPxMin) || (p->Px()>fPxMax) ||
128  (p->Py()<fPyMin) || (p->Py()>fPyMax) ||
129  (p->Pz()<fPzMin) || (p->Pz()>fPzMax) ) {
130  return kFALSE;
131  }
132  } else {
133  Double_t rapidity = 0.5*TMath::Log( (p->Energy()+p->Pz()) /
134  (p->Energy()-p->Pz()) );
135  if( (p->Pt()<fPtMin) || (p->Pt()>fPtMax) ||
136  (rapidity<fRapidityMin) || (rapidity>fRapidityMax) ) {
137  return kFALSE;
138  }
139  }
140 
141  // Apply energy cut
142  if( (p->Energy()<fEtotMin) || (p->Energy()>fEtotMax) ) {
143  return kFALSE;
144  }
145 
146  // Apply generation cut
147  if(-1 == p->GetFirstMother()) {
148  if(kFALSE == fStorePrim) {
149  return kFALSE;
150  }
151  } else {
152  if(kFALSE == fStoreSec) {
153  return kFALSE;
154  }
155  }
156 
157  return kTRUE;
158 }

162 void FairTrajFilter::SetVertexCut(Double_t vxMin, Double_t vyMin, Double_t vzMin,
163  Double_t vxMax, Double_t vyMax, Double_t vzMax)
164 {
165  TGeoBBox* cave = (TGeoBBox*) gGeoManager->GetTopVolume()->GetShape();
166  cave->ComputeBBox();
167  Double_t caveX = 2.*cave->GetDX();
168  Double_t caveY = 2.*cave->GetDY();
169  Double_t caveZ = 2.*cave->GetDZ();
170  if( (vxMax<vxMin) || (vyMax<vyMin) || (vzMax<vzMin) ||
171  (TMath::Abs(vxMin)>caveX) || (TMath::Abs(vxMax)>caveX) ||
172  (TMath::Abs(vyMin)>caveY) || (TMath::Abs(vyMax)>caveY) ||
173  (TMath::Abs(vzMin)>caveZ) || (TMath::Abs(vzMax)>caveZ) ) {
174  cout << "-E- FairTrajFilter::SetVertexCut() : invalid region, ignoring." << endl;
175  return;
176  }
177  fVxMin = vxMin;
178  fVxMax = vxMax;
179  fVyMin = vyMin;
180  fVyMax = vyMax;
181  fVzMin = vzMin;
182  fVzMax = vzMax;
183 }
