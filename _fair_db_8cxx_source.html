71 ValTimeStamp FairDb::MakeTimeStamp(const std::string& sqlDateTime,
72  Bool_t* ok)
73 {
74 
75  struct date {
76  int year;
77  int month;
78  int day;
79  int hour;
80  int min;
81  int sec;
82  };
83  char dummy;
84 
85  static string lo = "1970-01-01 00:00:00";
86  static string hi = "2038-01-19 03:14:07";
87 
88  // Set up defaults from 0:0am today.
89  ValTimeStamp nowTS;
90  int nowDate = nowTS.GetDate();
91  date defaultDate = {nowDate/10000, nowDate/100%100, nowDate%100,0,0,0};
92  date input = defaultDate;
93 
94  // istringstream in(sqlDateTime);
95  // in >> input.year >> dummy >> input.month >> dummy >> input.day
96  // >> input.hour >> dummy >> input.min >> dummy >> input.sec;
97 
98  sscanf(sqlDateTime.c_str(),"%d-%d-%d %d:%d:%d",&input.year,&input.month,&input.day,&input.hour,&input.min,&input.sec);
99 
100  if ( ok ) { *ok = kTRUE; }
101 
102  if ( sqlDateTime < lo || sqlDateTime > hi ) {
103  if ( ok ) { *ok = kFALSE; }
104  else {
105  static int bad_date_count = 0;
106  if ( ++bad_date_count <= 20 ) {
107  const char* last = (bad_date_count == 20) ? "..Last Message.. " : "";
108 
109  DBLOG("FairDb",FairDbLog::kError) << "Bad date string: " << sqlDateTime
110  << " parsed as "
111  << input.year << " "
112  << input.month << " "
113  << input.day << " "
114  << input.hour << " "
115  << input.min << " "
116  << input.sec
117  << "\n Outside range " << lo
118  << " to " << hi << last << endl;
119  }
120  }
121 
122  input = defaultDate;
123  }
124 
125  return ValTimeStamp(input.year,input.month,input.day,
126  input.hour,input.min,input.sec);
127 
128 }
