135 Double_t FairMCTrack::GetRapidity() const
136 {
137  Double_t e = GetEnergy();
138  Double_t y = 0.5 * TMath::Log( (e+fPz) / (e-fPz) );

163 // ----- Public method SetNPoints --------------------------------------
164 void FairMCTrack::SetNPoints(Int_t iDet, Int_t nPoints)
165 {
166 
167  if ( iDet == kREF ) {
168  if ( nPoints < 0 ) { nPoints = 0; }
169  else if ( nPoints > 1 ) { nPoints = 1; }
170  fNPoints = ( fNPoints & ( ~ 1 ) ) | nPoints;
171  }
172 
173  else if ( iDet == kTutDet ) {
174  if ( nPoints < 0 ) { nPoints = 0; }
175  else if ( nPoints > 7 ) { nPoints = 7; }
176  fNPoints = ( fNPoints & ( ~ ( 7 << 1 ) ) ) | ( nPoints << 1 );
177  }
178 
179  else if ( iDet == kFairRutherford ) {
180  if ( nPoints < 0 ) { nPoints = 0; }
181  else if ( nPoints > 31 ) { nPoints = 31; }
182  fNPoints = ( fNPoints & ( ~ ( 31 << 4 ) ) ) | ( nPoints << 4 );
183  }
184 
185  else LOG(ERROR) << "Unknown detector ID "
186  << iDet << FairLogger::endl;
187 
188 }
